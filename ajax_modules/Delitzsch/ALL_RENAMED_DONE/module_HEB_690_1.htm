לאגרת פטרוס השניה

אגרת יוחנן הראשונה
פרק א

דבר החיים

1 אֵת אֲשֶׁר הָיָה מֵרֹאשׁ אֲשֶׁר שָׁמַעְנוּ וּבְעֵינֵינוּ רָאִינוּ אֲשֶׁר הִבַּטְנוּ וַאֲשֶׁר מִשֲׁשׁוּ יָדֵינוּ עַל-דְּבַר הַחַיִּים׃ 2 וְהַחַיִּים נִגְלוּ וַנֵּרֶא וּמְעִידִים אֲנַחְנוּ וּמוֹדִיעִים לָכֶם אֶת-חַיֵּי הָעוֹלָמִים אֲשֶׁר הָיוּ עִם-הָאָב וְנִגְלוּ לָנוּ׃ 3 אֵת אֲשֶׁר רָאִינוּ וְשָׁמַעְנוּ נוֹדִיעָה לָכֶם (תהילים עח' ג'-ד') לְמַעַן תִּתְחַבְּרוּ לָנוּ גַּם-אַתֶּם וְנִתְחַבַּרְנוּ עִם-הָאָב וְעִם-בְּנוֹ יֵשׁוּעַ הַמָּשִׁיחַ׃ 4 וְזֹאת כֹּתְבִים אֲנַחְנוּ לָכֶם לְמַעַן תִּהְיֶה שִׂמְחַתְכֶם שְׁלֵמָה׃

האלהים הוא אור

5 וְזֹאת הַשְּׁמוּעָה אֲשֶׁר שָׁמַעְנוּ מִמֶּנּוּ וַנַּגֵּד לָכֶם שֶׁהָאֱלֹהִים אוֹר הוּא וְכָל-חֹשֶׁךְ אֵין בּוֹ׃(תהילים קד' ב') 6 אִם-נֹאמַר כִּי יֶשׁ-לָנוּ הִתְחַבְּרוּת עִמּוֹ וְנִתְהַלֵּךְ בַּחֹשֶׁךְ דֹּבְרֵי שֶׁקֶר אֲנַחְנוּ וְלֹא פֹּעֲלֵי הָאֱמֶת׃ 7 אֲבָל אִם-נִתְהַלֵּךְ בָּאוֹר כַּאֲשֶׁר הוּא בָאוֹר הִנֵּה נִתְחַבַּרְנוּ יָחַד וְדַם-יֵשׁוּעַ הַמָּשִׁיחַ בְּנוֹ יְטַהֲרֵנוּ מִכָּל-חֵטְא׃ 8 אִם-נֹאמַר שֶׁאֵין-בָּנוּ חֵטְא מַתְעִים אֲנַחְנוּ נַפְשֵׁנוּ וְהָאֱמֶת אֵין בָּנוּ׃(תהילים נא' ז') 9 וְאִם-נִתְוַדֶּה אֶת-חַטֹּאתֵינוּ נֶאֱמָן הוּא וְצַדִּיק לִסְלֹחַ לְחַטֹּאתֵינוּ וּלְטַהֲרֵנוּ מִכָּל-עָוֹן׃(משלי כח' יג'; תהילים לב' ה') 10 וְאִם-נֹאמַר שֶׁלּא חָטָאנוּ לְכֹזֵב נְשִׂימֶנּוּ וּדְבָרוֹ אֵין בָּנוּ׃
