אגרת שאול השניה אל התסלוניקים
פרק ג

נחמת המאמינים -- המשך

1 וְעוֹד אַחַי הִתְפַּלֲלוּ בַעֲדֵנוּ אֲשֶׁר יָרוּץ דְּבַר יְהוָֹה וְיִכָּבֵד כְּהִכָּבְדוֹ בְקִרְבְּכֶם׃ 2 וַאֲשֶׁר נִנָּצֵל מִן-הָאֲנָשִׁים הַתֹּעִים וְהָרָעִים (תהילים קמ' ב') כִּי לֹא כָל-אֲנָשִׁים אֱמוּנָה בָם׃ כִּי לֹא לְכָל-אָדָם הָאֱמוּנָה׃ 3 אֲבָל נֶאֱמָן הוּא (דברים ז' ט'; ישעיהו מט' ז') הוּא אֲשֶׁר יְעוֹדֵד אֶתְכֶם וְיִשְׁמָרְכֶם מִן-הָרָע׃ הָאָדוֹן אֲשֶׁר יְחַזֵּק אֶתְכֶם וְיִשְׁמָרְכֶם מִן-הָרָע׃ 4 וַאֲנַחְנוּ בְּטוּחִים עֲלֵיכֶם בַּאֲדֹנֵינוּ שֶׁעֹשִׂים אַתֶּם כְּמִצְוָתֵנוּ וְגַם-תּוֹסִיפוּ לַעֲשׂוֹת׃ 5 וְהָאָדוֹן הוּא יְיַשֵּׁר אֶת-לְבַבְכֶם לְאַהֲבַת הָאֱלֹהִים וּלְסַבְלָנוּת הַמָּשִׁיחַ׃(דברי הימים א', כט' יח')

החובה לעבוד

6 וְהִנְנוּ מְצַוִּים אֶתְכֶם אַחִים בְּשֵׁם אֲדֹנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ אֲשֶׁר תִּבָּדְלוּ מִכָּל-אָח מְעַקֵּשׁ דְּרָכָיו וְאֵינֶנּוּ מִתְהַלֵּךְ עַל-פִּי הַקַּבָּלָה אֲשֶׁר קִבֵּל מֵאִתָּנוּ׃ 7 הֲלֹא אַתֶּם יְדַעְתֶּם שֶׁצְּרִיכִים אַתֶּם לְהִתְהַלֵּךְ כָּמֹנוּ כִּי לֹא-הֶעֱוִינוּ אֶת-דַּרְכֵּנוּ בְּתוֹכֲכֶם׃ 8 גַּם לֹא-אָכַלְנוּ לֶחֶם אִישׁ חִנָּם כִּי בְּעָמָל וּתְלָאָה לַיְלָה וְיוֹמָם יָגַעְנוּ לְבִלְתִּי הֱיוֹת לְאִישׁ מִכֶּם לְמַשָׂא׃ 9 לֹא מִפְּנֵי שֶׁאֵין-לָנוּ הָרְשׁוּת הַזֹּאת כִּי אִם-לָתֵת אֹתָנוּ לָכֶם לְמוֹפֵת לָלֶכֶת בְּעִקְּבוֹתֵינוּ׃ 10 כִּי גַּם-בִּהְיוֹתֵנוּ אֶצְלְכֶם צִוִּינוּ אֶתְכֶם לֵאמֹר אֲשֶׁר לֹא יִרְצֶה לַעֲשׂוֹת גַּם-לֹא יֹאכֵל׃(בראשית ג' יח') 11 כִּי שָׁמַעְנוּ שֶׁיֵּשׁ בָּכֶם מְעַקְּשִׁים אָרְחֹתָם וְאֵינָם עֹשִׂים דָּבָר וּמְבַלִּים יְמֵיהֶם בַּהֲבָלִים׃ 12 וַאֲנָשִׁים כָּאֵלֶּה נְצַוֶּה אוֹתָם וְנַזְהִירֵם בַּאֲדֹנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ לַעֲבֹד בְּנַחַת לְמַעַן יֹאכְלוּ אֶת-לַחְמָם מִשֶּׁלָּהֶם׃ 3 וְאַתֶּם אֶחָי אַל-תִּתְרַפּוּ בַּעֲשׂוֹת הַטּוֹב׃ 14 וְאִם לֹא-יִשְׁמַע אִישׁ אֶל-דְּבָרֵנוּ בָּאִגֶּרֶת הַזֹּאת אֹתוֹ תִרְשְׁמוּ לָכֶם וְאַל-תִּתְעָרֲבוּ עִמּוֹ לְמַעַן יֵבוֹשׁ׃ 15 אַךְ לֹא-כְאֹיֵב תַּחְשְׁבֻהוּ כִּי אִם-תּוֹכִיחֻהוּ כְּאָח׃(ויקרא יט' יז') 16 וַאֲדוֹן הַשָּׁלוֹם הוּא יִתֵּן לָכֶם הַשָּׁלוֹם תָּמִיד (במדבר ו' כו') וַאֲדֹנֵינוּ יְהִי עִם-כֻּלְּכֶם׃ וּבְכָל-פָּנִים׃ 17 יְהִי הָאָדוֹן עִם-כֻּלְּכֶם׃ 18 שֹׁאֵל לִשְׁלוֹמְכֶם בִּכְתָב יָדִי אֲנִי פוֹלוֹס וְהִיא הָאוֹת בְּכָל-הָאִגְּרוֹת כֵּן אָנֹכִי כֹתֵב׃ 19 חֶסֶד יֵשׁוּעַ הַמָּשִׁיחַ אֲדֹנֵינוּ עִם-כֻּלְּכֶם׃
