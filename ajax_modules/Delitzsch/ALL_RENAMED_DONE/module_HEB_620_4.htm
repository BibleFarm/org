לפרק הקודם

אגרת שאול השניה אל טימותיאוס
פרק ד

התמדה בָּאמת -- המשך

1 וְעַתָּה הִנְנִי מֵעִיד בְּךָ נֶגֶד הָאֱלֹהִים וְנֶגֶד אֲדֹנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ הַבָּא לִשְׁפֹּט אֶת-הַחַיִּים וְאֶת-הַמֵּתִים בְּהוֹפָעָתוֹ וּבְמַלְכוּתוֹ׃ 2 הַכְרֵז אֶת-הַדָּבָר וְהַתְמֵד בְּעִתּוֹ וְשֶׁלּא בְעִתּוֹ הוֹכַח וּגְעַר וְהַזְהֵר בְּכָל-אֹרֶךְ-רוּחַ וְהוֹרָאָה׃ 3 כִּי-בוֹא תָבֹא הָעֵת אֲשֶׁר לֹא-יָכִילוּ אֶת-הַלֶּקַח הַבָּרִיא כִּי אִם-כְּאַוַּת נַפְשָׁם יַאַסְפוּ לָהֶם מוֹרִים לְשַׁעֲשֻׁעֵי אָזְנָיִם׃ 4 וְיַטּוּ אָזְנֵיהֶם מִן-הָאֱמֶת וְיִפְנוּ אֶל-הַהַגָּדוֹת׃ 5 אֲבָל אַתָּה הֱיֵה עֵר בַּכֹּל סְבֹל הָרָעוֹת וַעֲשֵׂה מְלֶאכֶת הַמְבַשֵׂר וּמַלֵּא אֶת-שֵׁרוּתֶךָ׃ 6 כִּי עַתָּה זֶה אֻסַּךְ נָסֶךְ וְעֵת פְּטִירָתִי הִגִּיעָה׃ 7 הַמִּלְחָמָה הַטּוֹבָה נִלְחָמְתִּי אֶת-הַמְּרוּצָה הִשְׁלַמְתִּי אֶת-הָאֱמוּנָה נָצָרְתִּי׃ 8 וּמֵעַתָּה שָׁמוּר לִי כֶּתֶר הַצְּדָקָה אֲשֶׁר בַּיּוֹם הַהוּא יִתֵּן לִי הָאָדוֹן הַשֹּׁפֵט הַצַּדִּיק (תהילים ט' ה') וְלֹא-לִי לְבַדִּי כִּי-לְכָל-אֹהֲבֵי הוֹפָעָתוֹ׃

בקשות אישיות

9 חוּשָׁה לָבוֹא אֵלַי בִּמְהֵרָה׃ 10 כִּי-דִימָס עֲזָבַנִי בְּאַהֲבָתוֹ אֶת-הָעוֹלָם הַזֶּה וַיֵּלֶךְ-לוֹ לְתַסְלוֹנִיקִי וּקְרִיסְקִיס הָלַךְ לְגָלַטְיָה וְטִיטוֹס לְדַלְמָטְיָה׃ 11 וְלוּקָס הוּא לְבַדּוֹ עִמָּדִי קַח אֶת-מַרְקוֹס וַהֲבִיאֵהוּ אִתָּךְ כִּי יוֹעִיל-לִּי לִמְאֹד לָעֲבוֹדָה׃ 12 אֶת-טוּכִיקוֹס שָׁלַחְתִּי לְאֶפְסוֹס׃ 13 אֶת-הַמִּטְפַּחַת שֶׁהִנַּחְתִּי בִטְרוֹאַס אֵצֶל קַרְפּוֹס הֲבִיאֵהוּ אִתְּךָ בְּבוֹאֶךָ וְאַף אֶת-הַסְּפָרִים וְעַל-כֻּלָּם אֶת-כִּתְבֵי הַקְּלָף׃ 14 אֲלֶכְסַנְדְּרוֹס חָרַשׁ הַנְּחשֶׁת עָשָׂה לִי רָעוֹת רַבּוֹת יְשַׁלֶּם-לוֹ יְהוָֹה כְּמַעֲשָׂיו׃(שמואל ב', ג' לט'; תהילים כח' ד') 15 וְגַם-אַתָּה הִשָּׁמֶר-לְךָ מִמֶּנּוּ כִּי-מָרָה אֶת-דְּבָרֵינוּ׃ 16 בְּהִצְטַדְּקִי בָרִאשׁוֹנָה לֹא-הָיָה אִישׁ לְעָזְרֵנִי כִּי-כֻלָּם עֲזָבוּנִי אַל-תֵּחָשֵׁב לָהֶם לְעָוֹן׃ 17 וַאדֹנָי הוּא עֲזָרַנִי וְחִזְּקָנִי לְמַעַן תֻּשְׁלַם עַל-יָדִי הַבְּשׂוֹרָה וְיִשְׁמָעוּהָ כָּל-הַגּוֹיִם וָאֶנָּצֵל מִפִּי אַרְיֵה׃(תהילים כב' כב'; דניאל ו' כח') 18 וְיַצִּילֵנִי אֲדֹנָי מִכָּל-מַעֲשֶׂה רָע (תהילים קכא' ז'; ירמיהו טו' כ'-כא') וְיוֹשִׁיעֵנִי לְמַלְכוּתוֹ בַשָּׁמָיִם לוֹ הַכָּבוֹד לְעוֹלְמֵי עוֹלָמִים אָמֵן׃ 19 שְׁאַל לִשְׁלוֹם פְּרִיסְקָה וַעֲקִילַס וּבֵית אֳנִיסִיפוֹרוֹס׃ 20 אֲרַסְטוֹס נִשְׁאַר בְּקוֹרִנְתּוֹס וְאֶת-טְרוֹפִימוֹס הִנַּחְתִּי בְּמִילִיטוֹס כִּי חֹלֶה הוּא׃ 21 חוּשָׁה לָבוֹא לִפְנֵי הַסְּתָיו אוֹבוּלוֹס וּפוּדִיס וְלִינוֹס וּקְלוֹדְיָה וְהָאַחִים כֻּלָּם שֹׁאֲלִים לִשְׁלוֹמֶךָ׃ 22 אֲדֹנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ יְהִי עִם-רוּחֶךָ הַחֶסֶד עִמָּכֶם אָמֵן׃
