אגרת אל העברים
פרק י

קרבן המשיח מסיר חטאים -- המשך

1 כִּי הַתּוֹרָה בִּהְיוֹת בָּה צֵל הַטֹּבוֹת הָעֲתִידוֹת וְלֹא מַרְאֵה עֶצֶם הַדְּבָרִים אֵין בִּיכָלְתָּהּ לְעוֹלָם לְהַשְׁלִים אֶת-הַקְּרֵבִים בַּקָּרְבָּנוֹת הָהֵם אֲשֶׁר יַקְרִיבוּ תָמִיד מִדֵּי שָׁנָה בְשָׁנָה׃ 2 כִּי אִם-כֵּן הֲלֹא חָדְלוּ לַהֲבִיאָם כִּי לֹא-הָיְתָה עוֹד יְדִיעַת חֲטָאִים בְּמַקְרִיבִים בְּהִטָּהֲרָם פַּעַם אֶחָת׃ 3 אֲבָל יֶשׁ-שָׁם הַזְכָּרַת הַחֲטָאִים שָׁנָה בְשָׁנָה׃(ויקרא טז' לד') 4 כִּי דַם-הַפָּרִים וְהַשְׂעִירִים לֹא יוּכַל לְהָסִיר חֲטָאִים׃(מיכה ו' ו'-ז'; ירמיהו יא' טו') 5 וְעַל-כֵּן אֹמֵר בְּבוֹאוֹ לָעוֹלָם זֶבַח וּמִנְחָה לֹא חָפַצְתָּ גְּוִיָּה כּוֹנַנְתָּ לִּי׃ 6 עוֹלָה וְחַטָּאָה לֹא שָׁאָלְתָּ׃(תהילים נא' יח') 7 אָז אָמַרְתִּי הִנֵּה-בָאתִי בִּמְגִלַּת-סֵפֶר כָּתוּב עָלַי לַעֲשׂוֹת רְצוֹנְךָ אֱלֹהָי׃(תהילים מ' ז'-ט') 8 אַחֲרֵי אָמְרוֹ לְמַעְלָה זֶבַח וּמִנְחָה עוֹלָה וְחַטָּאָה לֹא חָפַצְתָּ וְלֹא שָׁאָלְתָּ אֲשֶׁר יַקְרִיבוּ אֹתָם עַל-פִּי הַתּוֹרָה׃ 9 אָז אָמַר הִנֵּה-בָאתִי לַעֲשׂוֹת רְצוֹנְךָ אֱלֹהָי (***) מַעֲבִיר בָּזֶה אֶת-הָרִאשׁוֹן לְמַעַן הָקִים אֶת-הַשֵּׁנִי׃ 10 וּבָרָצוֹן הַזֶּה מְקֻדָּשִׁים אֲנַחְנוּ עַל-יְדֵי הַקְרָבַת קָרְבַּן גּוּפוֹ שֶׁל-יֵשׁוּעַ הַמָּשִׁיחַ בְּפַעַם אֶחָת׃ 11 וְכָל-כֹּהֵן עֹמֵד יוֹם יוֹם לְשָׁרֵת וּמוֹסִיף פְּעָמִים רַבּוֹת לְהַקְרִיב הַקָּרְבָּנוֹת הָהֵמָּה אֲשֶׁר לֹא-יוּכְלוּ לְעוֹלָם לְהַעֲבִיר חֲטָאִים׃(שמות כט' לח'//***) 12 וְהוּא אַחֲרֵי הַקְרִיבוֹ זֶבַח אֶחָד עַל-הַחֲטָאִים יָשַׁב לִימִין הָאֱלֹהִים לָנֶצַח׃(תהילים קי' א') 13 וּמֵאָז יְחַכֶּה עַד כִּי-יוּשְׁתוּ אֹיְבָיו הֲדֹם לְרַגְלָיו׃(***) 14 כִּי הוּא בְּקָרְבָּן אֶחָד הִשְׁלִים לָנֶצַח אֶת-הַמְקֻדָּשִׁים׃ 15 וְאַף-רוּחַ הַקֹּדֶשׁ מֵעִיד לָנוּ עַל-זֹאת כִּי אַחֲרֵי אָמְרוֹ׃ 16 זֹאת הַבְּרִית אֲשֶׁר אֶכְרֹת אִתָּם אַחֲרֵי הַיָּמִים הָהֵם אָמַר יְהוָֹה נָתַתִּי אֶת-תּוֹרָתִי בְּקִרְבָּם וְעַל-לִבָּם אֶכְתֳבֶנָּה׃(ירמיהו לא' לב'-לג') 17 וְלַעֲוֹנָם וּלְחַטָּאתָם לֹא אֶזְכָּר-עוֹד׃(***) 18 וְהִנֵּה בִּמְקוֹם שֶׁיֵּשׁ סְלִיחַת הַחֲטָאִים אֵין עוֹד קָרְבָּן עֲלֵיהֶם׃

דברי עידוד והזהרה

19 וְעַתָּה אֶחָי בִּהְיוֹת לָנוּ בִּטְחוֹן דֶּרֶךְ הַקֹּדֶשׁ בְּדַם יֵשׁוּעַ׃ 20 דֶּרֶךְ חָדָשׁ וָחָי אֲשֶׁר חִדֵּשׁ לָנוּ בַּפָּרֹכֶת הִיא בְשָׂרוֹ׃ 21 וּבִהְיוֹת לָנוּ כֹּהֵן גָּדוֹל עַל-בֵּית אֱלֹהִים׃(זכריה ו' יג') 22 נִקְרְבָה-נָּא בְּלֵבָב שָׁלֵם וּבֶאֱמוּנָה תְמִימָה מְטֹהָרִים בְּהַזָּיַת לְבָבֵנוּ מֵרוּחַ רָעָה וּרְחוּצֵי הַבָּשָׂר בְּמַיִם טְהוֹרִים׃(שמות כט' ד') 23 נַחֲזִיקָה בְּהוֹדָאַת הַתִּקְוָה בַּל-נִמּוֹט כִּי-נֶאֱמָן הַמַּבְטִיחַ׃ 24 וְנִתְבּוֹנֲנָה אִישׁ עַל-אָחִיו לְעוֹרֵר אֹתָנוּ לְאַהֲבָה וּלְמַעֲשִׂים טוֹבִים׃ 25 וְאַל-נַעֲזֹב אֶת-כְּנֵסִיָּתֵנוּ כְּדֶרֶךְ קְצָת אֲנָשִׁים כִּי אִם-נַזְהִיר אִישׁ אֶת-רֵעֵהוּ וּבְיוֹתֵר בִּרְאֹתְכֶם כִּי-קָרַב הַיּוֹם׃ 26 כִּי אִם-נֶחֱטָא בְזָדוֹן אַחֲרֵי אֲשֶׁר הָיְתָה לָּנוּ יְדִיעַת הָאֱמֶת לֹא-יִשָּׁאֵר עוֹד זֶבַח עַל-הַחֲטָאִים׃ 27 כִּי אִם-בִּעוּתֵי הַדִּין הֶעָתִיד וְאֵשׁ קִנְאָה אֲשֶׁר תֹּאכַל אֶת-הַצֹּרֲרִים׃(מלאכי ג' יט') 28 הֵן אִישׁ כִּי-יָפֵר תּוֹרַת משֶׁה מוֹת יָמוּת בְּלִי חֶמְלָה עַל-פִּי שְׁנַיִם עֵדִים אוֹ-שְׁלֹשָׂה׃(דברים יט' טו') 29 וּמַה-דַּעְתְּכֶם כַּמָה יִגְדַּל הָעֹנֶשׁ הַנָּכוֹן לְמִי שֶׁרֹמֵס בְּרַגְלוֹ אֶת-בֶּן-הָאֱלֹהִים וְעֹשֶׂה אֶת-דַּם הַבְּרִית אֲשֶׁר-הוּא מְקֻדָּשׁ בּוֹ כְּחֹל וּמְחָרֵף אֶת-רוּחַ הֶחָסֶד׃ 30 כִּי-יָדַעְנוּ מִי הוּא הָאֹמֵר לִי נָקָם וְשִׁלֵּם וְעוֹד כִּי-יָדִין יְהוָֹה עַמּוֹ׃(דברים לב' לה'-לו'; תהילים קלה' יד') 31 מַה-נּוֹרָא לִנְפֹּל בְּיַד אֱלֹהִים חַיִּים׃(דברים לב' לט'-מא') 32 אֲבָל זִכְרוּ-נָא אֶת-הַיָּמִים הָרִאשׁוֹנִים כִּי אָז אַחֲרֵי אֹרוּ עֵינֵיכֶם נְשָׂאתֶם צְבָא עִנּוּיִם רַבִּים׃ 33 פַּעַם בִּהְיוֹתְכֶם לְרַאֲוָה בְּחֶרְפָּה וְתוּגָה פַּעַם בְּהִשְׁתַּתֵּף עִם הַבָּאִים בַּצָּרוֹת כְּמוֹכֶם׃ 34 כִּי הִצְטָעַרְתֶּם עַל-הָאֲסִירִים וּגְזֵלַת רְכוּשְׁכֶם סְבַלְתֶּם בְּשִׂמְחָה מִדַּעְתְּכֶם בְּנַפְשְׁכֶם שֶׁיֵּשׁ-לָכֶם בַּשָּׁמַיִם קִנְיָן טוֹב מִמֶּנּוּ וְקַיָּם לָעַד׃ 35 לָכֵן אַל-תַּשְׁלִיכוּ אֶת-בִּטְחוֹנְכֶם כִּי יֶשׁ-לוֹ שָׂכָר רָב׃ 36 כִּי צְרִיכִים אַתֶּם לְסַבְלָנוּת לְמַעַן תַּעֲשׂוּ רְצוֹן אֱלֹהִים וּנְשָׂאתֶם אֶת-הַהַבְטָחָה׃ 37 כִּי עוֹד מְעַט-רָגַע וְהַבָּא יָבֹא לֹא יְאַחֵר׃ 38 הַצַּדִּיק בֶּאֱמוּנָתוֹ יִחְיֶה וְאִם-יִסֹּג אָחוֹר לֹא-רָצְתָה נַפְשִׁי בּוֹ׃(חבקוק ב' ג'-ד'; צפניה א' ד'-ו') 39 אֲבָל אֵין אֲנַחְנוּ מִן-הַנְּסֹגִים אָחוֹר לַאֲבֹד כִּי אִם-מִן-הַמַּאֲמִינִים לְהַצִּיל נַפְשֵׁנוּ׃
