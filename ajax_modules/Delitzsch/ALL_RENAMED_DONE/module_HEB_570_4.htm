לפרק הקודם

אגרת שאול אל הפיליפים
פרק ד

תמימות דעים, שמחה ושלום

1 עַל-כֵּן אַחַי הַחֲבִיבִים וְהַנֶּחְמָדִים שִׂמְחָתִי וַעֲטֶרֶת רֹאשִׁי עִמְדוּ-נָא כֵן בַּאֲדֹנֵינוּ חֲבִיבָי׃ 2 אֶת-אַבְהוֹדִיָּה אֲנִי מַזְהִיר וְאֶת-סוּנְטִיכִי אֲנִי מַזְהִיר לִהְיוֹת לֵב-אֶחָד בַּאֲדֹנֵינוּ׃ 3 וְאַתָּה חֲבֵרִי הַנִּצְמָד-לִי בֶאֱמֶת אַף-מִמְּךָ אֲנִי מְבַקֵּשׁ לִהְיוֹת לְעֵזֶר לְאֹתָם שֶׁיָּגְעוּ עִמָּדִי עַל-הַבְּשׂוֹרָה עִם-קְלִימִיס וְעִם-שְׁאָר תֹּמְכַי אֲשֶׁר שְׁמוֹתָם יַחַד בְּסֵפֶר הַחַיִּים׃(דניאל יב' א') 4 שִׂמְחוּ בַאֲדֹנֵינוּ בְּכָל-עֵת וְעוֹד הַפַּעַם אֹמֵר אֲנִי שְׂמָחוּ׃(תהילים פט' יז'; ישעיהו סא' י') 5 עַנְוַת רוּחֲכֶם תִּוָּדַע לְכָל-אִישׁ קָרֹב הוּא הָאָדוֹן׃ 6 אַל-תִּדְאָגוּ כִּי אִם-בִּתְפִלָּה וּבְתַחֲנוּנִים עִם-תּוֹדָה תָּבֹאנָה מִשְׁאֲלוֹתֵיכֶם לְכָל-דָּבָר לִפְנֵי אֱלֹהֵינוּ׃ 7 וּשְׁלוֹם אֱלֹהִים הַנַּעֲלֶה עַל-כָּל-שֵׂכֶל יִנְצֹר אֶת-לְבַבְכֶם וְאֶת-מַחְשְׁבוֹתֵיכֶם בַּמָּשִׁיחַ יֵשׁוּעַ׃ 8 סוֹף דָּבָר אַחַי כָּל-אֲשֶׁר הוּא אֱמֶת וְנִכְבָּד וְיָשָׁר וְטָהוֹר וְנָעִים וַאֲשֶׁר שָׁמְעוֹ טוֹב כָּל-מַעֲשֵׂה צֶדֶק כָּל-מַעֲשֵׂה שֶׁבַח עַל-אֵלֶּה תָּשִׂימוּ לְבַבְכֶם׃(תהילים יט' טו') 9 אֲשֶׁר לְמַדְתֶּם וְקִבַּלְתֶּם וּשְׁמַעְתֶּם וּרְאִיתֶם בִּי אֶת-אֵלֶּה תַעֲשׂוּ וֵאלֹהֵי הַשָּׁלוֹם יִהְיֶה עִמָּכֶם׃

השליח מודה לפיליפים על עזרתם

10 וַאֲנִי שָׂמַחְתִּי מְאֹד בַּאֲדֹנֵינוּ כִּי-עַתָּה הַפַּעַם הֶחֱלַפְתֶּם כֹּחַ לְהַשְׁגִּיחַ עָלָי וְגַם-מַשְׁגִּיחִים הֱיִיתֶם עַד-כֹּה אֶלָּא שֶׁלּא-עָלְתָה בְיֶדְכֶם׃ 11 וְלֹא מִמַּחְסוֹר אֲדַבֵּר כֵּן כִּי לָמַדְתִּי לְהִסְתַּפֵּק בְּמַה-שֶּׁיֵּשׁ לִי׃ 12 יֹדֵעַ אֲנִי לַעֲמֹד בְּעֹנִי וְיֹדֵעַ אֲנִי לְהוֹתִיר מֻרְגָּל אֲנִי לְכָל-דָּבָר וְאוֹפָן הֵן לִשְׂבֹּעַ הֵן לִרְעֹב הֵן לַעֲמֹד בְּשֶׁפַע הֵן לַחֲסֹר׃ 13 אֶת-כֹּל אוּכַל בְּעֶזְרַת [הַמָּשִׁיחַ] הַנּוֹתֵן לִי כֹּחַ׃ 14 אֲבָל הֵיטַבְתֶּם לַעֲשׂוֹת בְּהִתְחַבֶּרְכֶם אֵלַי בְּצָרָתִי׃ 15 וְיֹדְעִים גַּם-אַתֶּם פִילִיפִּיִּים שֶׁבְּרֵאשִׁית הַבְּשׂוֹרָה כְּצֵאתִי מִמַּקְדּוֹנְיָא לֹא-הִתְחַבְּרָה לִי אַחַת מִן-הַקְּהִלּוֹת לְמַשָׂא וּמַתָּן כִּי אִם-אַתֶּם בִּלְבָד׃ 16 כִּי-גַם בְּתַסְלוֹנִיקִי שְׁלַחְתֶּם לִי אֶת-צְרָכָי פַּעַם וּשְׁתָּיִם׃ 17 לֹא שֶׁאֲבַקֵּשׁ אֶת-הַמַּתָּן אֶלָּא אֲבַקֵּשׁ הַפְּרִי אֲשֶׁר-יִרְבֶּה בְּחֶשְׁבֹּנְכֶם׃ 18 וַאֲנִי קִבַּלְתִּי אֶת-הַכֹּל וְיֶשׁ-לִי דַּי וְהוֹתֵר וַאֲנִי נִמְלֵאתִי בְּקַבְּלִי מִידֵי אֶפַּפְרוֹדִיטוֹס אֶת-כֹּל אֲשֶׁר-שְׁלַחְתֶּם רֵיחַ נִיחֹחַ זֶבַח עָרֵב וְרָצוּי לֵאלֹהִים׃(ויקרא ב' ב') 19 וֵאלֹהַי הוּא יְמַלֵּא אֶת-כָּל-צָרְכְּכֶם כְּעשֶׁר כְּבוֹדוֹ בַּמָּשִׁיחַ יֵשׁוּעַ׃(תהילים לד' י') 20 וְלֵאלֹהִים אָבִינוּ הַכָּבוֹד לְעוֹלְמֵי עוֹלָמִים אָמֵן׃ 21 שַׁאֲלוּ לִשְׁלוֹם כָּל-קָדוֹשׁ בַּמָּשִׁיחַ יֵשׁוּעַ הָאַחִים אֲשֶׁר עִמִּי שֹׁאֲלִים לִשְׁלוֹמְכֶם׃ 22 כָּל-הַקְּדשִׁים שֹׁאֲלִים לִשְׁלוֹמְכֶם וּבְיוֹתֵר שֶׁל-בֵּית קֵיסָר׃ 23 חֶסֶד יֵשׁוּעַ הַמָּשִׁיחַ אֲדֹנֵינוּ [עִם רוּחֲכֶם] אָמֵן׃
