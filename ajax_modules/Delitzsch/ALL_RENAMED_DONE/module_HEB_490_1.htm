פרק א

הקדמה

1 אַחֲרֵי אֲשֶׁר רַבִּים הוֹאִילוּ לְחַבֵּר סִפּוּר הַמַּעֲשִׂים אֲשֶׁר נֶאֶמְנוּ בִשְׁלֵמוּת בְּתוֹכֵנוּ׃ 2 כַּאֲשֶׁר מְסָרוּם לָנוּ הָרֹאִים אֹתָם בְּעֵינֵיהֶם מִתְּחִלָּה וַאֲשֶׁר הָיוּ מְשָׁרֲתֵי הַדָּבָר׃ 3 חָשַׁבְתִּי לְטוֹב גַּם-אֲנִי הַחֹקֵר אַחַר כָּל-הַדְּבָרִים הֵיטֵב מֵרֵאשִׁיתָם לְכָתְבָם כְּסִדְרָם אֵלֶיךָ הָאַדִּיר תְּאוֹפִילוֹס׃ 4 לְמַעַן תֵּדַע אֱמֶת הָאֲמָרִים אֲשֶׁר לֻמָּדְתָּ׃

5 כֹּהֵן הָיָה בִּימֵי הוֹרְדוֹס מֶלֶךְ אֶרֶץ יְהוּדָה זְכַרְיָה שְׁמוֹ מִמִּשְׁמֶרֶת אֲבִיָּה (דברי הימים א', כד' י'; כד' יט'; נחמיה יב' יז') וְלוֹ אִשָּׁה מִבְּנוֹת אַהֲרֹן וּשְׁמָהּ אֱלִישָׁבַע׃ 6 וּשְׁנֵיהֶם צַדִּיקִים לִפְנֵי הָאֱלֹהִים וְהֹלְכֵי תֹם בְּכָל-מִצְוֹת יְהוָֹה וּבְחֻקֹּתָיו׃(בראשית ז' א'; דברים י' יב'; איוב א' א'; א' ח') 7 וְלָהֶם אֵין וָלָד כִּי אֱלִישֶׁבַע עֲקָרָה (בראשית יא' ל'; שמואל א', א' ה') וּשְׁנֵיהֶם בָּאוּ בַיָּמִים׃(בראשית יז' יז') 8 וַיְהִי הַיּוֹם בְּכַהֲנוֹ לִפְנֵי אֱלֹהִים בְּסֵדֶר מִשְׁמָרוֹ׃ 9 לְהַקְטִיר קְטֹרֶת לְפִי גוֹרָלוֹ כְּמִשְׁפַּט עֲבוֹדַת הַכֹּהֲנִים וַיָּבֹא אֶל-הֵיכַל יְהוָֹה׃(דברי הימים א', כד' ה'; שמות ל' א'; ל' ז') 10 וְכָל-קְהַל הָעָם מִתְפַּלֲּלִים בַּחוּץ בְּשַׁעַת הַקְּטֹרֶת׃ 11 וְהִנֵּה מַלְאַךְ יְהוָֹה נִרְאָה אֵלָיו עֹמֵד מִימִין מִזְבַּח הַקְּטֹרֶת׃ 12 וַיַּרְא זְכַרְיָה וַיִּבָּהֵל וְאֵימָה נָפְלָה עָלָיו׃ 13 וַיֹּאמֶר אֵלָיו הַמַּלְאָךְ אַל-תִּירָא זְכַרְיָהוּ (שופטים ו' כב'-כג') כִּי נִשְׁמְעָה תְּפִלָּתֶךָ (דניאל י' יב') וֶאֱלִישֶׁבַע אִשְׁתְּךָ תֵּלֵד לְךָ בֵּן וְקָרָאתָ שְׁמוֹ יוֹחָנָן׃(בראשית יז' יט') 14 וְהָיָה-לְךָ לְשִׂמְחָה וָגִיל וְרַבִּים יִשְׂמְחוּ בְּהִוָּלְדוֹ׃ 15 כִּי-גָדוֹל יִהְיֶה לִפְנֵי יְהוָֹה וְיַיִן וְשֵׁכָר לֹא יִשְׁתֶּה וְרוּחַ הַקֹּדֶשׁ יִמָּלֵא מִבֶּטֶן אִמּוֹ׃(במדבר ו' ג'; שופטים יג' ד'-ה') 16 וְרַבִּים מִבְּנֵי יִשְׂרָאֵל יָשִׁיב אֶל-יְהוָֹה אֱלֹהֵיהֶם׃ 17 וְהוּא יֵלֵךְ לְפָנָיו בְּרוּחַ אֵלִיָּהוּ וּבִגְבוּרָתוֹ (מלאכי ג' א') לְהָשִׁיב לֵב אָבוֹת עַל-בָּנִים (מלאכי ג' כד') וְאֵת הַסּוֹרֲרִים בִּתְבוּנַת הַצַּדִּיקִים לְהַעֲמִיד עַם מוּכָן לַיהוָֹה׃ 18 וַיֹּאמֶר זְכַרְיָה אֶל-הַמַּלְאָךְ בַּמָּה אֵדַע אֶת-הַדָּבָר הַזֶּה (בראשית טו' ח') כִּי-אֲנִי זָקַנְתִּי וְאִשְׁתִּי בָּאָה בַיָּמִים׃(***) 19 וַיַּעַן הַמַּלְאָךְ וַיֹּאמֶר אֵלָיו אֲנִי גַבְרִיאֵל הָעוֹמֵד לִפְנֵי הָאֱלֹהִים (דניאל ח' טז'; ט' כא') וְשָׁלוּחַ אָנֹכִי לְדַבֵּר אֵלֶיךָ וּלְבַשֶּׂרְךָ אֶת-הַבְּשׂוֹרָה הַזֹּאת׃(תהילים קג' כ') 20 וְהִנְּךָ נֶאֱלָם וְלֹא תוּכַל לְדַבֵּר עַד-הַיּוֹם אֲשֶׁר יָקוּם הַדָּבָר הַזֶּה תַּחַת כִּי-לֹא הֶאֱמַנְתָּ לִדְבָרַי וְהֵם יִמָּלְאוּ בְּמוֹעֲדָם׃ 21 וְהָעָם הוֹחִילוּ לִזְכַרְיָה וַיִּתְמְהוּ כִּי-הִתְמַהְמַהּ בַּהֵיכָל׃ 22 וַיְהִי בְצֵאתוֹ לֹא יָכֹל לְדַבֵּר אֲלֵיהֶם וַיֵּדְעוּ כִּי-מַרְאָה רָאָה בַּהֵיכָל וַיִּרְמֹז לָהֶם וְעוֹדֶנּוּ נֶאֱלָם׃ 23 וַיְהִי כַּאֲשֶׁר מָלְאוּ יְמֵי עֲבֹדָתוֹ וַיָּשָׁב אֶל-בֵּיתוֹ׃ 24 וַיְהִי אַחַר הַיָּמִים הָאֵלֶּה וַתַּהַר אֱלִישֶׁבַע אִשְׁתּוֹ וַתִּתְחַבֵּא חֲמִשָּׁה חֳדָשִׁים וַתֹּאמַר׃ 25 כָּכָה עָשָׂה לִי יְהוָֹה בִּימֵי פָקְדוֹ אוֹתִי לֶאֱסֹף אֶת-חֶרְפָּתִי מִבְּנֵי אָדָם׃(בראשית ל' כג')

מלאך אלהים מודיע מראש את הולדת ישוע

26 וַיְהִי בַּחֹדֶשׁ הַשִּׁשִּׁי וַיִּשְׁלַח אֱלֹהִים אֶת-גַּבְרִיאֵל (***) הַמַּלְאָךְ גָּלִילָה אֶל-עִיר אַחַת וּשְׁמָהּ נְצָרֶת׃ 27 אֶל-בְּתוּלָה מְאֹרָשָׂה לְאִישׁ אֲשֶׁר-שְׁמוֹ יוֹסֵף מִבֵּית דָּוִד וְשֵׁם הַבְּתוּלָה מִרְיָם׃ 28 וַיָּבֹא הַמַּלְאָךְ הַחַדְרָה וַיֹּאמֶר אֵלֶיהָ שָׁלוֹם לָךְ אֵשֶׁת-חֵן (שופטים ה' כד') יְהוָֹה עִמָּךְ [בְּרוּכָה אַתְּ בַּנָּשִׁים]׃ 29 וְהִיא [בִּרְאוֹתָהּ] נִבְהֲלָה לִדְבָרוֹ וַתֹּאמֶר בְּלִבָּהּ מָה הַבְּרָכָה הַזֹּאת׃ 30 וַיֹּאמֶר לָהּ הַמַּלְאָךְ אַל-תִּירְאִי מִרְיָם כִּי-מָצָאתְ חֵן לִפְנֵי הָאֱלֹהִים׃ 31 וְהִנָּךְ הָרָה וְיֹלַדְתְּ בֵּן וְקָרָאתְ אֶת-שְׁמוֹ יֵשׂוּעַ׃ 32 וְהוּא גָּדוֹל יִהְיֶה וּבֶן-עֶלְיוֹן יִקָּרֵא וַיהוָֹה אֱלֹהִים (תהילים ב' ז') יִתֶּן-לוֹ אֶת-כִּסֵּא דָּוִד אָבִיו׃ 33 וּמָלַךְ עַל-בֵּית יַעֲקֹב לְעוֹלָם וָעֶד וּלְמַלְכוּתוֹ אֵין קֵץ׃(תהילים קלב' יא'; דניאל ב' מד'; ז' כב') 34 וַתֹּאמֶר מִרְיָם אֶל-הַמַּלְאָךְ אֵיךְ יִהְיֶה הַדָּבָר הַזֶּה וַאֲנִי אֵינֶנִּי יֹדַעַת אִישׁ׃ 35 וַיַּעַן הַמַּלְאָךְ וַיֹּאמֶר אֵלֶיהָ רוּחַ הַקֹּדֶשׁ תָּבוֹא עָלַיךְ וּגְבוּרַת עֶלְיוֹן תָּצֵל עָלָיּךְ עַל-כֵּן קָדוֹשׁ יֵאָמֵר לַיִּלוֹד בֶּן-הָאֱלֹהִים׃(ישעיהו ז' יד'; ט' ו'; שמואל ב', ז' יב'-יג') 36 וְהִנֵּה אֱלִישֶׁבַע קְרוֹבָתֵךְ אֲשֶׁר קָרְאוּ-לָהּ עֲקָרָה גַּם-הִיא הָרָה לָלֶדֶת בֵּן בְּזִקְנָתָהּ וְזֶה לָּהּ הַחֹדֶשׁ הַשִּׁשִּׁי׃ 37 כִּי לֹא-יִפָּלֵא מֵאֱלֹהִים כָּל-דָּבָר׃(בראשית יח' יד'; ישעיהו נה' י'-יא') 38 וַתֹּאמֶר מִרְיָם הִנְנִי שִׁפְחַת יְהוָֹה יְהִי-לִי כִּדְבָרֶךָ וַיֵּצֵא מֵאִתָּהּ הַמַּלְאָךְ׃

מרים מבקרת אצל אלישבע

39 וַתָּקָם מִרְיָם בַּיָּמִים הָהֵם וַתְּמַהֵר לָלֶכֶת הָהָרָה אֶל-עִיר יְהוּדָה׃ 40 וַתָּבֹא בֵּית זְכַרְיָה וַתְּבָרֶךְ אֶת-אֱלִישָׁבַע׃ 41 וַיְהִי כִּשְׁמֹעַ אֱלִישֶׁבַע אֶת-בִּרְכַּת מִרְיָם וַיִּרְקַד הַיֶּלֶד בְּמֵעֶיהָ וַתִּמָּלֵא אֱלִישֶׁבַע רוּחַ הַקֹּדֶשׁ׃ 42 וַתִּקְרָא בְּקוֹל גָּדוֹל וַתֹּאמַר בְּרוּכָה אַתְּ בַּנָּשִׁים וּבָרוּךְ פְּרִי בִטְנֵךְ׃ 43 וּמַה-לִּי כִּי-אֵם אֲדֹנִי בָּאָה אֵלָי׃ 44 כִּי קוֹל בִּרְכָתֵךְ בָּא בְאָזְנָי וְהִנֵּה רָקַד בְּשִׂמְחָה הַיֶּלֶד בְּמֵעָי׃ 45 וְאַשְׁרֵי הַמַּאֲמִינָה כִּי הִמָּלֵא יִמָּלֵא אֲשֶׁר דֻּבַּר-לָהּ מֵאֵת יְהוָֹה׃ 46 וַתֹּאמֶר מִרְיָם רוֹמֲמָה נַפְשִׁי אֶת-יְהוָֹה׃ 47 וַתָּגֶל רוּחִי בֵּאלֹהֵי יִשְׁעִי׃(תהילים לה' ט'; חבקוק ג' יח') 48 אֲשֶׁר רָאָה בָּעֳנִי אֲמָתוֹ (שמואל א', א' יא') כִּי הִנֵּה מֵעַתָּה יְאַשְּׁרוּנִי כָּל-הַדֹּרוֹת׃ 49 כִּי גְדֹלוֹת עָשָׂה לִי שַׁדָּי (בראשית יז' א'; דברים י' כא'; תהילים קכו' ג') וְקָדוֹשׁ שְׁמוֹ׃(תהילים קיא' ט') 50 וְחַסְדּוֹ לְדוֹר דּוֹרִים עַל יְרֵאָיו׃(תהילים קג' יז') 51 גְּבוּרוֹת עָשָׂה בִּזְרֹעוֹ פִּזַּר גֵּאִים בִּמְזִמּוֹת לִבָּם׃(שמות טו' ו'; תהילים פט' יא') 52 הָרַס נְדִיבִים מִכִּסְאוֹתָם וַיָּרֶם שְׁפָלִים׃(תהילים קמז' ו'; איוב ה' יא') 53 רְעֵבִים מִלֵּא-טוֹב וַעֲשִׁירִים שִׁלַּח רֵיקָם׃(תהילים קז' ט'; לד' יא'; שמואל ב', ה' ז'-ח') 54 תָּמַךְ בְּיִשְׂרָאֵל עַבְדּוֹ לִזְכֹּר אֶת-רַחֲמָיו׃(ישעיהו מא' ח'; תהילים צח' ג') 55 כַּאֲשֶׁר דִּבֶּר אֶל-אֲבוֹתֵינוּ לְאַבְרָהָם וּלְזַרְעוֹ עַד-עוֹלָם׃(מיכה ז' כ'; בראשית יז' ז') 56 וַתֵּשֶׁב מִרְיָם עִמָּהּ כִּשְׁלשָׁה חֳדָשִׁים וַתָּשָׁב לְבֵיתָהּ׃

הולדת יוחנן המטביל

57 וַיִּמְלְאוּ יְמֵי אֱלִישֶׁבַע לָלֶדֶת וַתֵּלֶד בֵּן׃ 58 וַיִּשְׁמְעוּ שְׁכֵנֶיהָ וּקְרוֹבֶיהָ כִּי-הִגְדִּיל יְהוָֹה אֶת-חַסְדּוֹ עִמָּהּ וַיִּשְׂמְחוּ אִתָּה׃ 59 וַיְהִי בַּיּוֹם הַשְּׁמִינִי וַיָּבֹאוּ לָמוּל אֶת-הַיָּלֶד (בראשית יז' יב') וַיִּקְרְאוּ אֶת-שְׁמוֹ זְכַרְיָה עַל-שֵׁם אָבִיו׃ 60 וַתַּעַן אִמּוֹ וַתֹּאמַר לֹא כִּי יוֹחָנָן יִקָּרֵא לוֹ׃ 61 וַיֹּאמְרוּ אֵלֶיהָ אֵין-אִישׁ בְּמִשְׁפַּחְתֵּךְ אֲשֶׁר שְׁמוֹ כַּשֵּׁם הַזֶּה׃ 62 וַיִּרְמְזוּ אֶל-אָבִיו לָדַעַת מָה הַשֵּׁם אֲשֶׁר יִקָּרֵא לוֹ׃ 63 וַיִּשְׁאַל לוּחַ וַיִּכְתֹּב עָלָיו לֵאמֹר יוֹחָנָן שְׁמוֹ וַיִּתְמְהוּ כֻּלָּם׃ 64 וַיִּפָּתַח פִּיו וּלְשׁוֹנוֹ פִּתְאֹם וַיְדַבֵּר וַיְבָרֶךְ אֶת-הָאֱלֹהִים׃(תהילים נא' יז') 65 וַתִּפֹּל אֵימָה עַל-כָּל-שְׁכֵנֵיהֶם וַיְסֻפַּר כָּל-הַדְּבָרִים הָאֵלֶּה בְּכָל-הָרֵי יְהוּדָה׃ 66 וַיָּשִׂימוּ כָל-הַשֹּׁמְעִים אֶל-לִבָּם לֵאמֹר מָה-אֵפוֹא יִהְיֶה הַיֶּלֶד הַזֶּה וְיַד-יְהוָֹה הָיְתָה עִמּוֹ׃ 67 וַיִּמָּלֵא זְכַרְיָה אָבִיו רוּחַ הַקֹּדֶשׁ וַיִּנָּבֵא לֵאמֹר׃ 68 בָּרוּךְ יְהוָֹה אֱלֹהֵי יִשְׂרָאֵל כִּי (תהילים מא' יד'; עב' יח') פָקַד אֶת-עַמּוֹ וַיִּשְׁלַח לוֹ פְּדוּת׃(תהילים קיא' ט') 69 וַיַּצְמַח לָנוּ קֶרֶן יְשׁוּעָה בְּבֵית דָּוִד עַבְדּוֹ׃(תהילים קלב' יז'; שמואל א', ב' י') 70 כַּאֲשֶׁר דִּבֶּר בְּפִי-נְבִיאָיו הַקְּדוֹשִׁים אֲשֶׁר מֵעוֹלָם׃ 71 יְשׁוּעָה מֵאֹיְבֵינוּ וּמִיַּד כָּל-שׂנְאֵינוּ׃(תהילים מד' ח'; קו' י') 72 לַעֲשׂוֹת חֶסֶד עִם-אֲבוֹתֵינוּ (מיכה ז' כ') וְלִזְכֹּר אֶת-בְּרִית קָדְשׁוֹ׃(תהילים קה' ח'-ט'; קו' מה'; בראשית יז' ז') 73 אֶת-הַשְּׁבוּעָה אֲשֶׁר נִשְׁבַּע לְאַבְרָהָם אָבִינוּ׃(בראשית כב' טז'; ירמיהו יא' ה') 74 לְהַצִּילֵנוּ מִיַּד אֹיְבֵינוּ וּלְתִתֵּנוּ לְעָבְדוֹ בְּלִי-פָחַד׃ 75 בְּתָמִים וּבִצְדָקָה לְפָנָיו כָּל-יְמֵי חַיֵּינוּ׃ 76 וְאַתָּה הַיֶּלֶד נְבִיא עֶלְיוֹן יִקָּרֵא לָךְ כִּי לִפְנֵי יְהוָֹה תֵּלֵךְ לְפַנּוֹת אֶת-דְּרָכָיו׃(מלאכי ג' א') 77 וּלְהוֹרוֹת דֶּרֶךְ הַיְשׁוּעָה לְעַמּוֹ בִּסְלִיחַת חַטֹּאתֵיהֶם׃(ירמיהו לא' לג') 78 בְּחֶסֶד אֱלֹהֵינוּ וּבְרַחֲמָיו אֲשֶׁר בָּהֶם יִפְקְדֵנוּ הַנֹּגַהּ מִמָּרוֹם׃(ישעיהו ס' א'-ב'; סג' ז'; מלאכי ג' כ') 79 לְהָאִיר לְישְׁבֵי חשֶׁךְ וְצַלְמָוֶת (ישעיהו ט' א'; תהילים קז' י'; קז' יד') וּלְהָכִין אֶת-רַגְלֵינוּ אֶל-דֶּרֶךְ הַשָּׁלוֹם׃ 80 וַיִּגְדַּל הַיֶּלֶד וַיֶּחֱזַק בָּרוּחַ וַיְהִי בַמִּדְבָּרוֹת עַד-יוֹם הֵרָאֹתוֹ אֶל-יִשְׂרָאֵל׃
