פרק כ

אלף השנים

1 וָאֵרֶא מַלְאָךְ יוֹרֵד מִן-הַשָּׁמָיִם וּבְיָדוֹ מַפְתֵּחַ הַתְּהוֹם וְכֶבֶל גָּדוֹל׃ 2 וַיִּתְפֹּשׂ אֶת-הַתַּנִין אֶת-הַנָּחָשׁ הַקַּדְמוֹנִי הוּא הַמַּלְשִׁין וְהוּא הַשָׂטָן (בראשית ג' א'; ג' יד') וַיַּאַסְרֵהוּ אֶלֶף שָׁנִים׃ 3 וַיַּשְׁלִיכֵהוּ אֶל-תְּהוֹם וַיִּסְגֹּר עָלָיו וַיַּחְתֹּם עָלָיו לְמַעַן אֲשֶׁר לֹא-יַדִּיחַ עוֹד אֶת-הַגּוֹיִם עַד-מְלֹאת אֶלֶף הַשָּׁנִים וְאַחֲרֵי-כֵן יֻתַּר לִזְמַן מִצְעָר׃ 4 וָאֵרֶא כִסְאוֹת וַיֵּשְׁבוּ עֲלֵיהֶם וְהַמִּשְׁפָּט נִתַּן בְּיָדָם (דניאל ז' ט'; ז' כב') וְנַפְשׁוֹת הַהֲרוּגִים עַל-עֵדוּת יֵשׁוּעַ וְעַל-דְּבַר הָאֱלֹהִים וַאֲשֶׁר לֹא הִשְׁתַּחֲווּ לַחַיָּה וּלְצַלְמָהּ וְלֹא קִבְּלוּ אֶת-תָּוָהּ עַל-מִצְחוֹתָם וְעַל-יָדָם וַיִּחְיוּ וַיִּמְלְכוּ עִם-הַמָּשִׁיחַ אֶלֶף שָׁנִים׃ 5 וְיֶתֶר הַמֵּתִים לֹא-חָיוּ עַד-מְלֹאת אֶלֶף הַשָּׁנִים זֹאת הַתְּחִיָּה הָרִאשׁוֹנָה׃ 6 וּמִי שֶׁיֶּשׁ-לוֹ חֵלֶק בַּתְּחִיָּה הָרִאשׁוֹנָה אַשְׁרָיו וְקָדוֹשׁ הוּא בָּהֶם לֹא-יִשְׁלֹט הַמָּוֶת הַשֵּׁנִי כִּי יִהְיוּ כֹּהֲנִים לֵאלֹהִים וְלִמְשִׁיחוֹ וּמָלְכוּ אִתּוֹ אֶלֶף שָׁנִים׃(ישעיהו סא' ו')

מפלת השטן

7 וְאַחֲרֵי כְלוֹת אֶלֶף הַשָּׁנִים יֻתַּר הַשָׂטָן מִבֵּית מִשְׁמָרוֹ׃ 8 וְיָצָא לְהַדִּיחַ אֶת-הַגּוֹיִם בְּאַרְבַּע כַּנְפוֹת הָאָרֶץ אֶת-גּוֹג וּמָגוֹג (יחזקאל לח' ב'; בראשית י' ב') וּלְקַבְּצָם לַמִּלְחָמָה אֲשֶׁר מִסְפָּרָם כְּחוֹל הַיָּם׃ 9 וַיַּעֲלוּ עַל-מֶרְחֲבֵי אֶרֶץ וַיָּסֹבּוּ אֶת-מַחֲנֵה הַקְּדֹשִׁים (דניאל ז' כא') וְאֶת-הָעִיר הַחֲבִיבָה וַתֵּרֶד אֵשׁ [מֵאֵת הָאֱלֹהִים] מִן-הַשָּׁמַיִם וַתֹּאכַל אֹתָם׃(יחזקאל לח' כב'; לט' ו') 10 וְהַשָׂטָן אֲשֶׁר הִדִּיחָם הֻשְׁלַךְ בַּאֲגַם-אֵשׁ וְגָפְרִית אֲשֶׁר-שָׁם גַּם-הַחַיָּה וּנְבִיא הַשָּׁקֶר וְיֻסְּרוּ יוֹמָם וָלַיְלָה לְעוֹלְמֵי עוֹלָמִים׃(ישעיהו לד' י')

המשפט האחרון

11 וָאֵרֶא כִּסֵּא לָבָן וְגָדוֹל וְהַיֹּשֵׁב עָלָיו מִפָּנָיו נָסוּ אֶרֶץ וְשָׁמָיִם וְלֹא-נִמְצָא לָהֶם מָקוֹם׃(דניאל ז' ט'; ישעיהו ו' א'; לד' ד'; נא' ו') 12 וָאֵרֶא אֶת-הַמֵּתִים הַקְּטַנִּים עִם-הַגְּדֹלִים עֹמְדִים לִפְנֵי הַכִּסֵּא וּסְפָרִים נִפְתָּחִים וַיִּפָּתַח סֵפֶר אַחֵר וְהוּא סֵפֶר הַחַיִּים (דניאל ז' י'; יב' א') וַיִּשָּׁפְטוּ הַמֵּתִים מִן-הַכָּתוּב בַּסְּפָרִים כְּמַעֲשֵׂיהֶם׃(משלי כד' יב'; קוהלת יב' יד') 13 וַיָּשֶׁב הַיָּם אֶת-מֵתָיו וְהַמָּוֶת וְהַשְּׁאוֹל הֵשִׁיבוּ אֶת-מֵתֵיהֶם וַיִּשָּׁפְטוּ אִישׁ אִישׁ כְּמַעֲשֵׂיהֶם׃ 14 וְהַמָּוֶת וְהַשְּׁאוֹל הֻשְׁלְכוּ בַאֲגַם-הָאֵשׁ וְהוּא הַמָּוֶת הַשֵּׁנִי׃ 15 וְכָל-אִישׁ אֲשֶׁר לֹא-נִמְצָא כָתוּב בְּסֵפֶר הַחַיִּים הֻשְׁלַךְ בַּאֲגַם הָאֵשׁ׃(דניאל יב' א'; ז' יא')
