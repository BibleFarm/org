לפרק הקודם

אגרת שאול אל האפסים
פרק ו

יחסים בין הבריות במשפחה ובחברה -- המשך

1 שִׁמְעוּ בָּנִים אֶל-הוֹרֵיכֶם בַּאֲדֹנֵינוּ (משלי ו' כ') כִּי-אֹרַח ישֶׁר הוּא׃ 2 כַּבֵּד אֶת-אָבִיךָ וְאֶת-אִמֶּךָ (דברים ה' טו'; שמות כ' יא') זֹאת הִיא הַמִּצְוָה הָרִאשׁוֹנָה שֶׁמַּתַּן שְׂכָרָהּ בְּצִדָּהּ׃ 3 לְמַעַן יִיטַב לָךְ וּלְמַעַן יַאֲרִיכֻן יָמֶיךָ עַל-הָאֲדָמָה׃(***) 4 וְאַתֶּם הָאָבוֹת אַל-תַּכְעִיסוּ אֶת-בְּנֵיכֶם וְגַדְּלוּ אֹתָם בְּמוּסַר אֲדֹנֵינוּ וּבְתוֹכַחְתּוֹ׃(משלי ג' יא'-יב') 5 שִׁמְעוּ עֲבָדִים אֶל-אֲדֹנֵיכֶם מִן-הַבָּשָׂר בְּיִרְאָה וָרֶתֶת וּבְתָם-לְבַבְכֶם כִּשְׁמֹעַ לַמָּשִׁיחַ׃ 6 לֹא בַעֲבֹדָה שֶׁהִיא לְמַרְאֵה עֵינַיִם כִּמְבַקְשִׁים לִמְצֹא חֵן בְּעֵינֵי בְנֵי-אָדָם כִּי אִם-כְּעַבְדֵי הַמָּשִׁיחַ הָעֹשִׂים רְצוֹן הָאֱלֹהִים בְּכָל-נַפְשָׁם׃ 7 לַעֲבֹד בְּנֶפֶשׁ חֲפֵצָה אֶת-הָאָדוֹן וְלֹא בְנֵי אָדָם׃ 8 וִידַעְתֶּם כִּי כָּל-דָּבָר טוֹב אֲשֶׁר עָשָׂה הָאָדָם יְשַׁלְּמֶנּוּ לוֹ אֲדֹנֵינוּ בֵּין שֶׁהוּא-עֶבֶד וּבֵין שֶׁהוּא בֶן-חוֹרִין׃ 9 וְאַתֶּם הָאֲדֹנִים כָּכָה תַעֲשׂוּ גַּם-לָהֶם חִדְלוּ מִגְּעָר-בָּם וִידַעְתֶּם כִּי גַם-עֲלֵיכֶם יֵשׁ אָדוֹן בַּשָּׁמַיִם וְאֵין לְפָנָיו מַשּׂא פָנִים׃(דברי הימים ב', יט' ז')

המאבק נגד כוחות הרע

10 סוֹף דָּבָר אֶחָי חִזְקוּ בַאֲדֹנֵינוּ וּבְעֹז גְּבוּרָתוֹ׃ 11 לִבְשׁוּ כְּלֵי-נֶשֶׁק אֱלֹהִים לְמַעַן תּוּכְלוּ עֲמֹד נֶגֶד נִכְלֵי הַשָׂטָן׃ 12 כִּי לֹא עִם-בָּשָׂר וָדָם מִלְחָמָה לָנוּ כִּי עִם-שָׂרִים וְשַׁלִּיטִים עִם-הַמּשְׁלִים בְּחֶשְׁכַּת הָעוֹלָם הַזֶּה עִם-הָרוּחוֹת הָרָעוֹת אֲשֶׁר בַּמְּרוֹמִים׃ 13 עַל-כֵּן אֶחֱזוּ בִּכְלֵי-נֶשֶׁק אֱלֹהִים לְמַעַן תּוּכְלוּ לָקוּם בַּיּוֹם הָרָע וּלְמַלֵּא צְבָאֲכֶם וְלַעֲמוֹד׃ 14 עִמְדוּ-נָא חֲגוּרֵי הָאֱמֶת (שמות יב' יא'; ישעיהו יא' ה') בְּמָתְנֵיכֶם וּלְבוּשִׁים שִׁרְיוֹן הַצֶּדֶק׃(ישעיהו נט' יז') 15 וְרַגְלֵיכֶם מֻנְעָלוֹת לִהְיוֹת מוּכָנִים לִבְשׂוֹרַת הַשָּׁלוֹם׃(ישעיהו נב' ז') 16 וְעַל-כָּל-אֵלֶּה תִּפְשׂוּ אֶת-מָגֵן הָאֱמוּנָה אֲשֶׁר תּוּכְלוּ לְכַבּוֹת בּוֹ אֶת-חִצֵּי הָרָשָׁע הַדֹּלְקִים׃ 17 וּקְחוּ אֶת-כּוֹבַע הַיְשׁוּעָה וְחֶרֶב הָרוּחַ שֶׁהִיא דְּבַר הָאֱלֹהִים׃ 18 וְהַתְמִידוּ בְכָל-עֵת בִּתְפִלָּה וְתַחֲנוּנִים בָּרוּחַ וְשִׁקְדוּ וְהִתְאַמְּצוּ לְהַעְתִּיר יַחַד בְּעַד כָּל-הַקְּדשִׁים׃ 19 וְגַם-בַּעֲדִי לְמַעַן יִנָּתֶן-לִי הַדָּבָר בְּפִתְחִי פִי לְגַלּוֹת בְּאֹמֶץ לֵב אֶת-סוֹד הַבְּשׂוֹרָה אֲשֶׁר אֲנִי מְלִיצָהּ בְּמוֹסֵרוֹת׃ 20 לְמַעַן אֲדַבֵּר בָּהּ כַּאֲשֶׁר מֻטָּל עָלַי לְדַבֵּר וְלֹא אֵחָת׃ 21 וּלְמַעַן אֲשֶׁר תֵּדְעוּ גַם-אַתֶּם אֶת-קֹרֹתַי וְאֶת-מַעֲשָׂי הִנֵּה טוּכִיקוֹס הָאָח הֶחָבִיב וְהַמְשָׁרֵת הַנֶּאֱמָן בַּאֲדֹנֵינוּ יַגִּיד לָכֶם הַכֹּל׃ 22 אֲשֶׁר לָזֹאת שְׁלַחְתִּיהוּ אֲלֵיכֶם לְהוֹדִיעֲכֶם אֶת-קֹרֹתֵינוּ וּלְמַעַן יְנַחֵם אֶת-לְבַבְכֶם׃ 23 הַשָּׁלוֹם לָאַחִים וְהָאַהֲבָה עִם-הָאֱמוּנָה מֵאֵת הָאֱלֹהִים הָאָב וּמֵאֵת הָאָדוֹן יֵשׁוּעַ הַמָּשִׁיחַ׃ 24 הַחֶסֶד עִם-כָּל-הָאֹהֲבִים אֶת-אֲדֹנֵינוּ יֵשׁוּעַ הַמָּשִׁיחַ בְּאַהֲבָה אֲשֶׁר לֹא-תִכְלֶה אָמֵן׃
