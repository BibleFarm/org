פרק ב

המכתב הראשון

1 אֶל-מַלְאַךְ קְהַל אֶפְסוֹס כְּתֹב כֹּה אָמַר הָאֹחֵז בִּימִינוֹ שִׁבְעַת הַכּוֹכָבִים הַמִּתְהַלֵּךְ בְּתוֹךְ שֶׁבַע מְנֹרוֹת הַזָּהָב׃(זכריה ד' ב') 2 יָדַעְתִּי אֶת-מַעֲשֶׂיךָ וְאֶת-עֲמָלְךָ וְאֶת-סָבְלֶךָ וַאֲשֶׁר לֹא-תוּכַל שְׂאֵת אֶת-הָרְשָׁעִים וַתִּבְחַן אֶת-הָאֹמְרִים שְׁלִיחִים אֲנַחְנוּ וְאֵינָם וַתִּמְצָאֵם דֹּבְרֵי שָׁקֶר׃(תהילים קלט' כא') 3 וְאַתָּה נָשָׂאתָ הַרְבֵּה וְיֶשׁ-לְךָ סַבְלָנוּת וּלְמַעַן שְׁמִי לֹא יָעָפְתָּ׃ 4 אַךְ יֶשׁ-לִי עָלֶיךָ כִּי נָטַשְׁתָּ אֶת-אַהֲבָתְךָ הָרִאשׁוֹנָה׃ 5 זָכְרָה אֵפוֹא אֵי-מִזֶּה נָפַלְתָּ וְשׁוּבָה וַעֲשֵׂה כְמַעֲשֶׂיךָ הָרִאשֹׁנִים וְאִם-לֹא הִנְנִי בָא עָלֶיךָ מַהֵר וְהָדַפְתִּי מְנוֹרָתְךָ מִמְּקוֹמָהּ אִם-לֹא תָשׁוּב׃(ירמיהו יח' ח') 6 אֲבָל זֹאת לָךְ אֲשֶׁר שׂנֵא אַתָּה אֶת-מַעֲשֵׂי הַנִּיקְלָסִים אֲשֶׁר שְׂנֵאתִים גַּם-אָנִי׃ 7 מִי אֲשֶׁר-אֹזֶן לוֹ יִשְׁמַע אֵת אֲשֶׁר הָרוּחַ אֹמֵר לַקְּהִלּוֹת הַמְנַצֵּחַ אֶתֶּן-לוֹ לֶאֱכֹל מֵעֵץ הַחַיִּים אֲשֶׁר בְּתוֹךְ גַּן-אֱלֹהִים׃(בראשית ב' ט'; ג' כב')

המכתב השני

8 וְאֶל-מַלְאַךְ קְהַל זְמִירְנָא כְּתֹב כֹּה אָמַר הָרִאשׁוֹן וְהָאַחֲרוֹן אֲשֶׁר-מֵת וַיֶּחִי׃(ישעיהו מד' ו'; מח' יב') 9 יָדַעְתִּי אֶת-מַעֲשֶׂיךָ וְאֶת-צָרָתְךָ וְאֶת-רֵישְׁךָ וְאוּלָם עָשִׁיר אָתָּה וְאֶת-גִּדּוּפֵי הָאֹמְרִים יְהוּדִים אֲנַחְנוּ וְאֵינָם כִּי אִם-כְּנֵסֶת הַשָּׂטָן׃ 10 אַל-תִּירָא מִפְּנֵי-אֲשֶׁר עָלֶיךָ לִסְבֹּל הִנֵּה עָתִיד הַמַּלְשִׁין לָתֵת מִכֶּם בַּמִּשְׁמָר לְמַעַן תְּנֻסּוּ וִהְיִיתֶם בַּצָּרָה עֲשֶׂרֶת יָמִים הֱיֵה נֶאֱמָן עַד-הַמָּוֶת וְנָתַתִּי לְךָ עֲטֶרֶת הַחַיִּים׃ 11 מִי אֲשֶׁר-אֹזֶן לוֹ יִשְׁמַע אֵת אֲשֶׁר הָרוּחַ אֹמֵר לַקְּהִלּוֹת הַמְנַצֵּחַ לֹא יִנָּזֵק בַּמָּוֶת הַשֵּׁנִי׃

המכתב השלישי

12 וְאֶל-מַלְאַךְ קְהַל פַּרְגְּמוֹס כְּתֹב כֹּה אָמַר אֲשֶׁר-לוֹ חֶרֶב פִּיפִיּוֹת הַחַדָּה׃ 13 יָדַעְתִּי אֶת-מַעֲשֶׂיךָ וְאֶת-מְקוֹם שִׁבְתְּךָ אֲשֶׁר-שָׁם כִּסֵּא הַשָּׂטָן וַתִּדְבַּק בִּשְׁמִי וְלֹא שִׁקַּרְתָּ בֶּאֱמוּנָתִי גַּם-בִּימֵי אַנְטִיפַס עֵדִי הַנֶּאֱמָן אֲשֶׁר נֶהֱרַג אֶצְלְכֶם בִּמְקוֹם מוֹשַׁב הַשָּׂטָן׃ 14 אַךְ-מְעַט יֶשׁ-לִי עָלֶיךָ כִּי-שָׁם עִמְּךָ אֲנָשִׁים דְּבֵקִים בְּלֶקַח בִּלְעָם אֲשֶׁר הוֹרָה אֶת-בָּלָק לָתֵת מִכְשׁוֹל לִפְנֵי בְּנֵי יִשְׂרָאֵל לֶאֱכֹל מִזִּבְחֵי אֱלִילִים וְלִזְנוֹת׃(במדבר כה' א'-ב'; לא' טז') 15 כֵּן נִמְצְאוּ גַּם-בְּךָ אֲנָשִׁים דְּבֵקִים בְּלֶקַח הַנִּיקְלָסִים אֲשֶׁר שָׂנֵאתִי׃ 16 שׁוּבָה וְאִם-לֹא תָשׁוּב אָבוֹא עָלֶיךָ מְהֵרָה וְנִלְחַמְתִּי בָם בְּחֶרֶב פִּי׃ 17 מִי אֲשֶׁר-אֹזֶן לוֹ יִשְׁמַע אֵת אֲשֶׁר הָרוּחַ אֹמֵר לַקְּהִלּוֹת הַמְנַצֵּחַ אַאֲכִילֶנּוּ מִן-הַמָּן הַגָּנוּז (תהילים עח' כג'-כד') וְנָתַתִּי לוֹ אֶבֶן לְבָנָה וְעַל-הָאֶבֶן מְפֻתָּח שֵׁם חָדָשׁ אֲשֶׁר לֹא יֵדָעֶנּוּ אִישׁ זוּלָתִי אֲשֶׁר נִתַּן לוֹ׃(ישעיהו נו' ה'; סב' ב')

המכתב הרביעי

18 וְאֶל-מַלְאַךְ קְהַל תִּיאֲטִירָא כְּתֹב כֹּה אָמַר בֶּן-הָאֱלֹהִים (תהילים ב' ז') אֲשֶׁר עֵינָיו כְּלַבַּת-אֵשׁ וּמַרְגְּלוֹתָיו כְּעֵין-נְחשֶׁת קָלָל׃(דניאל י' ו'; יחזראל א' כז') 19 יָדַעְתִּי אֶת-מַעֲשֶׂיךָ וְאַהֲבָתְךָ וֶאֱמוּנָתְךָ וַעֲבוֹדָתְךָ וְסָבְלֶךָ וַאֲשֶׁר מַעֲשֶׂיךָ הָאַחֲרוֹנִים רַבִּים מִן-הָרִאשׁוֹנִים׃ 20 אַךְ-מְעַט יֶשׁ-לִי עָלֶיךָ כִּי תַנִּיחַ לָאִשָּׁה לְאִיזֶבֶל (מלכים א', טז' לא') הָאֹמֶרֶת נְבִיאָה אָנֹכִי לְלַמֵּד וּלְהַתְעוֹת אֶת-עֲבָדַי לִזְנוֹת וְלֶאֱכֹל זִבְחֵי אֱלִילִים׃(במדבר כה' א'-ב'; מלכים ב', ט' כב') 21 וָאֶתֶּן-לָהּ זְמַן לָשׁוּב וְהִיא מֵאֲנָה לָשׁוּב מִתַּזְנוּתֶיהָ׃ 22 הִנְנִי מַפִּיל אֹתָהּ לְמִשְׁכָּב וְאֶת-הַמְנָאֲפִים עִמָּהּ בְּצָרָה גְדוֹלָה אִם-לֹא יָשׁוּבוּ מִמַּעֲשֵׂיהֶם׃ 23 וְהָרַגְתִּי בַמָוֶת בָּנֶיהָ וְיָדְעוּ כָל-הַקְּהִלּוֹת כִּי אֲנִי חֹקֵר כְּלָיוֹת וָלֵב (תהילים ז' י'; ירמיהו יא' כ'; יז' י') וְנָתַתִּי לָכֶם לְאִישׁ כִּפְרִי מַעֲלָלָיו׃(תהילים סב' יג') 24 וְאַתֶּם שְׁאָר הַנִּמְצְאִים בְּתִיאֲטִירָא כֹּל אֲשֶׁר-אֵין לָהֶם הַלֶּקַח הַהוּא וְלֹא יָדְעוּ אֶת-עֲמֻקּוֹת הַשָּׂטָן כַּאֲשֶׁר הֵם מִתְהַלֲלִים לָכֶם אֲנִי אֹמֵר לֹא-אָשִׂים עֲלֵיכֶם מַשָּׂא אַחֵר׃ 25 אֶפֶס מַה-שֶּׁיֵּשׁ לָכֶם בּוֹ הַחֲזִיקוּ עַד-בֹּאִי׃ 26 וְהַמְנַצֵּחַ וְשֹׁמֵר אֶת-מַעֲשַׂי עַד-עֵת קֵץ אֶתֶּן-לוֹ שָׁלְטָן עַל-הַגּוֹיִם׃ 27 וְרָעָם בְּשֵׁבֶט בַּרְזֶל כִּכְלִי יוֹצֵר יְנֻפָּצוּ (תהילים ב' ח'-ט') כַּאֲשֶׁר קִבַּלְתִּי גַם-אָנֹכִי מֵאֵת אָבִי׃ 28 וְנָתַתִּי לוֹ כּוֹכַב הַשָּׁחַר׃(ישעיהו נח' ח'; שיר השירים ו' י'; ישעיהו יד' יב') 29 מִי אֲשֶׁר-אֹזֶן לוֹ יִשְׁמַע אֵת-אֲשֶׁר הָרוּחַ אֹמֵר לַקְּהִלּוֹת׃
