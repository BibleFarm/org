לפרק הקודם

התגלות
פרק יט

1 אַחֲרֵי-כֵן שָׁמַעְתִּי קוֹל גָּדוֹל כְּקוֹל הָמוֹן רַב בַּשָּׁמַיִם הָאֹמְרִים הַלֲלוּיָהּ הַיְשׁוּעָה וְהַכָּבוֹד וְהָעֹז לֵאלֹהֵינוּ׃(תהילים צג' א') 2 כִּי-אֱמֶת וְיָשָׁר מִשְׁפָּטָיו (תהילים קיט' קלז') כִּי-הֵשִׁיב גְּמוּל עַל-הַזּוֹנָה הַגְּדוֹלָה אֲשֶׁר-הִשְׁחִיתָה אֶת-הָאָרֶץ בְּתַזְנֻתָהּ לִדְרֹשׁ מִיָּדָהּ אֶת-דַּם-עֲבָדָיו׃(מלכים ב', ט' ז') 3 וַיֹּאמְרוּ שֵׁנִית הַלֲלוּיָהּ וַעֲשָׁנָהּ יַעֲלֶה לְעוֹלְמֵי עוֹלָמִים׃(ישעיהו לד' י') 4 וְעֶשְׂרִים וְאַרְבָּעָה הַזְּקֵנִים וְאַרְבַּע הַחַיּוֹת נָפְלוּ עַל-פְּנֵיהֶם וַיִּשְׁתַּחֲווּ לֵאלֹהִים הַיֹּשֵׁב עַל-הַכִּסֵּא וַיֹּאמְרוּ אָמֵן הַלֲלוּיָהּ׃ 5 וְקוֹל יוֹצֵא מִן-הַכִּסֵּא וַיֹּאמַר הַלֲלוּ אֶת-אֱלֹהֵינוּ כָּל-עֲבָדָיו וִירֵאָיו הַקְּטַנִּים עִם-הַגְּדוֹלִים׃(תהילים קלה' א'; כב' כד') 6 וָאֶשְׁמַע קוֹל כְּקוֹל הָמוֹן רַב וּכְקוֹל מַיִם רַבִּים (יחזקאל א' כד') וּכְקוֹל רְעָמִים חֲזָקִים וַיֹּאמְרוּ הַלֲלוּיָהּ כִּי-מָלַךְ אֱלֹהֵינוּ יְהוָֹה צְבָאוֹת׃ 7 נִשְׂמְחָה וְנָגִילָה וְנִתְּנָה לוֹ הַכָּבוֹד כִּי בָאָה חֲתֻנַּת הַשֶׂה וְאִשְׁתּוֹ הִתְקַדָּשָׁה׃(תהילים קיח' כד') 8 וַיִּנָּתֶן לָהּ לִלְבֹּשׁ בּוּץ טָהוֹר וָצַח כִּי הַבּוּץ הוּא צִדְקוֹת הַקְּדֹשִׁים׃(תהילים קלב' ט'; ישעיהו סא' י') 9 וַיֹּאמֶר אֵלַי כְּתֹב אַשְׁרֵי הַקְּרוּאִים אֶל-מִשְׁתֵּה חֲתֻנַּת הַשֶׂה וַיֹּאמֶר אֵלַי אֵלֶּה הַדְּבָרִים אֱמֶת הֵם דִּבְרֵי אֱלֹהִים׃ 10 וָאֶפֹּל לְרַגְלָיו לְהִשְׁתַּחֲוֹת לוֹ וַיֹּאמֶר אֵלַי רְאֵה אַל-תַּעֲשֶׂה-זֹּאת עֶבֶד אֲנִי כָּמוֹךָ וְחָבֵר לְּךָ וּלְאַחֶיךָ אֲשֶׁר עֵדוּת יֵשׁוּעַ בְּפִיהֶם הִשְׁתַּחֲוֵה לֵאלֹהִים (דברים ו' יג'; תהילים כט' ב') כִּי עֵדוּת יֵשׁוּעַ הִיא רוּחַ הַנְּבוּאָה׃

הרוכב על הסוס הלבן

11 וָאֵרֶא אֶת-הַשָּׁמַיִם נִפְתָּחִים וְהִנֵּה-סוּס לָבָן וְהָרֹכֵב עָלָיו יִקָּרֵא לוֹ נֶאֱמָן וַאֲמִתִּי וּבְצֶדֶק הוּא שֹׁפֵט וְלֹחֵם׃(תהילים מה' ה'; צו' יג'; ישעיהו יא' ד'-ה') 12 וְעֵינָיו כְּלַבַּת-אֵשׁ וַעֲטָרוֹת הַרְבֵּה בְּרֹאשׁוֹ (דניאל י' ו') וְלוֹ שֵׁם כָּתוּב אֲשֶׁר לֹא-יָדַע אִישׁ כִּי אִם-הוּא לְבַדּוֹ׃(משלי לד' ד') 13 וְהוּא לָבוּשׁ לְבוּשׁ מְאָדָּם בַּדָּם וּשְׁמוֹ נִקְרָא דְּבַר הָאֱלֹהִים׃(ישעיהו סג' א'-ג') 14 וְצִבְאוֹת הַשָּׁמַיִם יֹצְאִים אַחֲרָיו עַל-סוּסִים לְבָנִים וְהֵם מְלֻבָּשִׁים בִּגְדֵי-בוּץ לָבָן וְטָהוֹר׃ 15 וּמִפִּיו יֹצֵאת חֶרֶב חַדָּה לְהַכּוֹת אֶת-הַגּוֹיִם וְהוּא יִרְעֵם בְּשֵׁבֶט בַּרְזֶל (ישעיהו יא' ד') וְהוּא דֹרֵךְ פּוּרַת יֵין חֲמַת-אַף אֱלֹהֵי הַצְּבָאוֹת׃(ישעיהו סג' ג') 16 וְעַל-בִּגְדוֹ וְעַל-יְרֵכוֹ כָּתוּב שֵׁם מֶלֶךְ הַמְּלָכִים וַאֲדֹנֵי הָאֲדֹנִים׃ 17 וָאֵרֶא מַלְאָךְ אֶחָד עֹמֵד בַּשָּׁמֶשׁ וַיִּזְעַק קוֹל גָּדוֹל וַיֹּאמֶר אֶל-צִפּוֹר כָּל-כָּנָף אֲשֶׁר-תָּעוּף בַּחֲצִי הַשָּׁמַיִם בֹּאוּ וְהֵאָסְפוּ עַל-זֶבַח הַגָּדוֹל אֲשֶׁר לֵאלֹהִים׃ 18 וַאֲכַלְתֶּם בְּשַׂר מְלָכִים וּבְשַׂר שָׂרֵי אֲלָפִים וּבְשַׂר גִּבּוֹרִים וּבְשַׂר סוּסִים וְרֹכְבֵיהֶם וּבְשַׂר כָּל-בְּנֵי חוֹרִים וַעֲבָדִים הַקְּטַנִּים עִם-הַגְּדוֹלִים׃(יחזקאל לט' ד'; לט' יז'-כ') 19 וָאֵרֶא אֶת-הַחַיָּה וּמַלְכֵי הָאָרֶץ וַאֲגַפֵּיהֶם נִקְהָלִים לַעֲשׂוֹת מִלְחָמָה עִם-הָרֹכֵב עַל-הַסּוּס וְעִם-צְבָאוֹ׃(תהילים ב' ב') 20 וַתִּתָּפֵשׂ הַחַיָּה וּנְבִיא הַשֶּׁקֶר אִתָּהּ אֲשֶׁר-עָשָׂה הָאוֹתוֹת לְפָנֶיהָ וְהִדִּיחַ בָּהֶן אֶת-נֹשְׂאֵי תָּו-הַחַיָּה וְהַמִּשְׁתַּחֲוִים לְצַלְמָהּ וּשְׁנֵיהֶם הֻשְׁלְכוּ חַיִּים בַּאֲגַם-הָאֵשׁ הַבֹּעֵר בְּגָפְרִית׃(דניאל ז' יא'; ישעיהו ל' לג') 21 וְהַנִּשְׁאָרִים נֶהֶרְגוּ בַחֶרֶב הַיּוֹצֵאת מִפִּי-הָרֹכֵב עַל-הַסּוּס וְכָל-הָעוֹף שָׂבְעוּ מִבְּשָׂרָם׃
