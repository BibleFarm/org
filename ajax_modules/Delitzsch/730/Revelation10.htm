פרק י

המלאך והספר הקטן

1 וָאֵרֶא מַלְאָךְ אַחֵר אַבִּיר יוֹרֵד מִן הַשָּׁמַיִם (תהילים קג' כ') וְהוּא עֹטֶה עָנָן וְעַל-רֹאשׁוֹ כְּמַרְאֵה הַקֶּשֶׁת אֲשֶׁר בֶּעָנָן וּפָנָיו כַּשֶּׁמֶשׁ וְרַגְלָיו כְּעַמּוּדֵי אֵשׁ׃(דניאל י' ו'; יחזקאל א' כז'-כח') 2 וּבְיָדוֹ סֵפֶר קָטֹן פָּתוּחַ וַיָּשֶׂם אֶת-רֶגֶל יְמִינוֹ עַל-הַיָּם וְאֶת-שְׂמֹאלוֹ עַל-הָאָרֶץ׃(יחזקאל ב' ט') 3 וַיִּקְרָא בְּקוֹל גָּדוֹל כַּאֲשֶׁר יִשְׁאַג הָאַרְיֵה וּבְקָרְאוֹ דִּבְּרוּ שִׁבְעַת הָרְעָמִים בְּקוֹלֹתֵיהֶם׃(הושע יא' י') 4 וּכְדַבֵּר שִׁבְעַת הָרְעָמִים [בְּקוֹלֹתֵיהֶם] חָפַצְתִּי לִכְתֹּב וָאֶשְׁמַע קוֹל מִן-הַשָּׁמַיִם לֵאמֹר חֲתוֹם אֵת אֲשֶׁר-דִּבְּרוּ שִׁבְעַת הָרְעָמִים וְאַל-תִּכְתֹּב זֹאת׃(דניאל ח' כו'; דניאל יב' ד'; יב' ט') 5 וְהַמַּלְאָךְ אֲשֶׁר-רְאִיתִיו עֹמֵד עַל-הַיָּם וְעַל-הָאָרֶץ הֵרִים יְמִינוֹ אֶל-הַשָּׁמָיִם׃ 6 וַיִּשָּׁבַע בְּחֵי עוֹלְמֵי הָעוֹלָמִים (דניאל יב' ז') אֲשֶׁר-בָּרָא אֶת-הַשָּׁמַיִם וְכָל-אֲשֶׁר בָּהֶם וְהָאָרֶץ וְכָל-אֲשֶׁר בָּהּ וְהַיָּם וְכָל-אֲשֶׁר-בּוֹ שֶׁאֵין עוֹד שָׁהוּת׃(תהילים קמו' ו') 7 אֲבָל בִּימֵי קוֹל הַמַּלְאָךְ הַשְּׁבִיעִי בְּעָמְדוֹ לִתְקוֹעַ וְנִשְׁלַם סוֹד הָאֱלֹהִים כַּאֲשֶׁר בִּשֵׂר אֶת-עֲבָדָיו הַנְּבִיאִים׃(עמוס ג' ז') 8 וְהַקּוֹל אֲשֶׁר שָׁמַעְתִּי מִן-הַשָּׁמַיִם דִּבֶּר אֵלַי שֵׁנִית לֵאמֹר לֵךְ וְקַח אֶת-הַסֵּפֶר הַקָּטֹן הַפָּתוּחַ אֲשֶׁר בְּיַד-הַמַּלְאָךְ הָעֹמֵד עַל-הַיָּם וְעַל-הָאָרֶץ׃ 9 וָאָבֹא אֶל-הַמַּלְאָךְ וָאֹמַר אֵלָיו תְּנָה-לִּי אֶת-הַסֵּפֶר וַיֹּאמֶר אֵלַי קַח וֶאֱכֹל אֹתוֹ וְהוּא יֵמַר בְּבִטְנְךָ וּבְפִיךָ יִהְיֶה מָתוֹק כִּדְבָשׁ׃(יחזקאל ב' ח'; ג' א'-ג') 10 וָאֶקַּח אֶת-הַסֵּפֶר מִיַּד-הַמַּלְאָךְ וָאֹכְלֵהוּ וַיְהִי בְּפִי כִּדְבַשׁ לְמָתוֹק וְאַחֲרֵי אָכְלִי אֹתוֹ וַתִּמָּלֵא בִטְנִי מְרוֹרִים׃ 11 וַיֹּאמְרוּ אֵלָי עוֹד תָּשׁוּב תִּנָּבֵא עַל-עַמִּים וְגוֹיִם וּלְשֹׁנוֹת וּמְלָכִים רַבִּים׃
