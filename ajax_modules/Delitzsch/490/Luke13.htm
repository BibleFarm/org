הבשורה על-פי לוקס
פרק יג

חזרה בתשובה תציל מאבדון

1 וַיָּבֹאוּ אֲנָשִׁים בָּעֵת הַהִיא וַיַּגִּידוּ לוֹ אֶת-דְּבַר הַגְּלִילִים אֲשֶׁר עֵרַב פִּילָטוֹס דָּמָם עִם-זִבְחֵיהֶם׃ 2 וַיַּעַן יֵשׁוּעַ וַיֹּאמֶר אֲלֵיהֶם הַאֹמְרִים אַתֶּם שֶׁהַגְּלִילִים הָאֵלֶּה הָיוּ חַטָּאִים מִכָּל-אַנְשֵׁי הַגָּלִיל כִּי מְצָאָם כַּדָּבָר הַזֶּה׃ 3 לֹא כִּי-אֹמֵר אֲנִי לָכֶם אִם-לֹא תָשׁוּבוּ גַּם-אַתֶּם כֻּלְּכֶם תֹּאבֵדוּ׃ 4 אוֹ שְׁמֹנָה הֶעָשָׂר הָהֵם אֲשֶׁר נָפַל עֲלֵיהֶם הַמִּגְדָּל בְּשִׁלּחַ וַיְמִיתֵם הַאֹמְרִים אַתֶּם שֶׁהָיוּ אֲשֵׁמִים מִכָּל-הָאֲנָשִׁים הַיּשְׁבִים בִּירוּשָׁלָיִם׃ 5 לֹא כִּי-אֹמֵר אֲנִי לָכֶם אִם-לֹא תָשׁוּבוּ גַּם-אַתֶּם כֻּלְּכֶם תֹּאבֵדוּ׃(יחזקאל יח' ל'-לב')

משל התאנה הבלתי פוריה

6 וַיִּשָׂא מְשָׁלוֹ וַיֹּאמַר אִישׁ אֶחָד הָיְתָה-לּוֹ תְאֵנָה נְטוּעָה בְכַרְמוֹ וַיָּבֹא לְבַקֶּשׁ-בָּהּ פְּרִי וְלֹא מָצָא׃ 7 וַיֹּאמֶר אֶל-הַכֹּרֵם הִנֵּה-זֶה שָׁלשׁ שָׁנִים אָנֹכִי בָא לְבַקֵּשׁ פְּרִי בַּתְּאֵנָה הַזֹּאת וְלֹא מָצָאתִי כְּרֹת אוֹתָהּ לָמָּה-זֶּה תַשְׁחִית אֵת הָאֲדָמָה׃ 8 וַיַּעַן וַיֹּאמֶר אֵלָיו אֲדֹנִי הַנִּיחָה אֹתָהּ עוֹד הַשָּׁנָה הַזֹּאת עַד אֲשֶׁר-עֲזַקְתִּיהָ וְשַׂמְתִּי דֹמֶן לָהּ׃ 9 אוּלַי תַּעֲשֶׂה פֶרִי וְאִם-לֹא תַעֲשֶׂה וְכָרַתָּ אֹתָהּ בַּשָּׁנָה הָאֲחֶרֶת׃(ישעיהו ה' ה'-ו')

ריפוי בעלת המום ביום השבת

10 וַיְהִי הוּא מְלַמֵּד בְּיוֹם הַשַּׁבָּת בְּבֵית כְּנֵסֶת אֶחָד׃ 11 וְהִנֵּה אִשָּׁה אֲשֶׁר בָּהּ רוּחַ-חֳלִי כִּשְׁמֹנֶה עֶשְׂרֵה שָׁנָה וְהִיא כְפוּפָה וְלֹא יָכְלָה לָקוּם קוֹמָה זְקוּפָה׃ 12 וַיַּרְא יֵשׁוּעַ וַיִּקְרָא אֵלֶיהָ וַיֹּאמֶר לָהּ אִשָּׁה הֵחָלְצִי מֵחָלְיֵךְ׃(תהילים לג' ט') 13 וַיָּשֶׂם יָדָיו עָלֶיהָ וּכְרֶגַע קָמָה וְתִּתְעוֹדָד וַתְּשַׁבַּח אֶת-הָאֱלֹהִים׃(תהילים קמו' ח') 14 וַיִּכְעַס רֹאשׁ הַכְּנֵסֶת עַל-אֲשֶׁר רִפָּא יֵשׁוּעַ בַּשַּׁבָּת וַיַּעַן וַיֹּאמֶר אֶל-הָעָם שֵׁשֶׁת יָמִים הֵם אֲשֶׁר תֵּעָשֶׂה בָהֶם מְלָאכָה בָּאֵלֶּה בֹּאוּ וְהֵרָפְאוּ וְלֹא בְּיוֹם הַשַּׁבָּת׃(שמות כ' ט') 15 וַיַּעַן הָאָדוֹן וַיֹּאמֶר אֵלָיו הֶחָנֵף הֲלֹא יַתִּיר אִישׁ מִכֶּם בַּשַּׁבָּת אֶת-שׁוֹרוֹ אוֹ אֶת-חֲמֹרוֹ מִן-הָאֵבוּס וְיוֹלִיכֵהוּ לְהַשְׁקֹתוֹ׃ 16 וְזֹאת אֲשֶׁר הִיא בַּת-אַבְרָהָם וַאֲשֶׁר הַשָּׂטָן אֲסָרָהּ זֶה שְׁמֹנֶה עֶשְׂרֵה שָׁנָה הֲלֹא תֻתַּר מִמּוֹסְרוֹתֶיהָ בְּיוֹם הַשַּׁבָּת׃(תהילים קטז' טז') 17 וַיְהִי כְּאָמְרוֹ אֶת-הַדְּבָרִים הָאֵלֶּה נִכְלְמוּ כָּל-מִתְקוֹמֲמָיו וַיִּשְׂמַח כָּל-הָעָם עַל-כָּל-הַנִּפְלָאוֹת הַנַּעֲשׂוֹת עַל-יָדוֹ׃(תהילים קז' כא')

משלים על גרגיר החרדל ועל השאור

(מתי יג' 33-31;מרקוס ד' 32-30)

18 וַיֹּאמַר לְמָה דוֹמָה מַלְכוּת הָאֱלֹהִים וְאֶל-מָה אַמְשִׁילֶנָּה׃ 19 דּוֹמָה הִיא לְגַרְגַּר שֶׁל-חַרְדָּל אֲשֶׁר לְקָחוֹ אִישׁ וַיְשִׂימֵהוּ בְגַנּוֹ וַיִּצְמַח וַיְהִי לְעֵץ גָּדוֹל וְעוֹף הַשָּׁמַיִם יְקַנֵּן בַּעֲנָפָיו׃ 20 וַיֹּאמֶר עוֹד אֶל-מָה אֲדַמֶּה אֵת מַלְכוּת הָאֱלֹהִים׃ 21 דּוֹמָה הִיא לִשְׂאֹר אֲשֶׁר לְקָחַתּוּ אִשָּׁה וַתִּטְמְנֵהוּ בִּשְׁלשׁ סְאִים קֶמַח עַד אֲשֶׁר-יֶחְמַץ כֻּלּוֹ׃

הפתח הצר

(מתי ז' 14-13, 23-21)

22 וַיַּעֲבֹר בֶּעָרִים וּבַכְּפָרִים עָבוֹר וְלַמֵּד וַיֵּלֶךְ לְדַרְכּוֹ לָבוֹא יְרוּשָׁלָיִם׃ 23 וַיִּשְׁאָלֵהוּ אִישׁ לֵאמֹר אֲדֹנֵינוּ הַמְעַט הֵם הַנּוֹשָׁעִים׃ 24 וַיֹּאמֶר אֲלֵיהֶם הִתְאַמְּצוּ לָבוֹא בַּפֶּתַח הַצָּר כִּי-אֹמֵר אֲנִי לָכֶם רַבִּים יְבַקְשׁוּ לָבוֹא וְלֹא יוּכָלוּ׃ 25 מִיּוֹם אֲשֶׁר יָקוּם בַּעַל הַבַּיִת וְסָגַר אֶת-הַדָּלֶת וְתָחֵלּוּ לַעֲמֹד בַּחוּץ וְלִדְפֹּק עַל-הַדֶּלֶת לֵאמֹר אֲדֹנֵינוּ פְּתַח-לָנוּ וְעָנָה וְאָמַר אֲלֵיכֶם אֵינֶנִּי יוֹדֵעַ אֶתְכֶם מֵאַיִן אַתֶּם׃(בראשית ז' טז') 26 אָז תָּחֵלּוּ לֵאמֹר הֲלֹא אָכַלְנוּ וְשָׁתִינוּ לְפָנֶיךָ וּבִרְחֹבוֹתֵינוּ לִמָּדְתָּ׃ 27 וְיֹאמַר אֲנִי אֹמֵר לָכֶם אֵינֶנִּי יוֹדֵעַ אֶתְכֶם מֵאַיִן אַתֶּם סוּרוּ מִמֶּנִּי כָּל-פֹּעֲלֵי הָאָוֶן׃(תהילים ו' ט') 28 וְשָׁם תִּהְיֶה הַיְלָלָה וַחֲּרֹק הַשִּׁנָיִם כַּאֲשֶׁר תִּרְאוּ אֶת-אַבְרָהָם וְיִצְחָק וְיַעֲקֹב וְאֶת-כָּל-הַנְּבִיאִים בְּמַלְכוּת הָאֱלֹהִים וְאַתֶּם מְגֹרָשִׁים הַחוּצָה׃(תהילים קא' ד') 29 וְיָבֹאוּ מִמִּזְרָח וּמִמַּעֲרָב וּמִצָּפוֹן וּמִדָּרוֹם וְיָסֵבּוּ בְּמַלְכוּת הָאֱלֹהִים׃(ישעיהו מט' יב') 30 וְהִנֵּה יֵשׁ אַחֲרוֹנִים אֲשֶׁר יִהְיוּ רִאשׁוֹנִים וְרִאשׁוֹנִים אֲשֶׁר יִהְיוּ אַחֲרוֹנִים׃

קינה על ירושלים

(מתי כג' 39-37)

31 בַּיּוֹם הַהוּא נִגְּשׁוּ מִן-הַפְּרוּשִׁים וַיֹּאמְרוּ אֵלָיו צֵא וְלֵךְ מִזֶּה כִּי הוֹרְדוֹס מְבַקֵּשׁ לְהָרְגֶךָ׃ 32 וַיֹּאמֶר אֲלֵיהֶם לְכוּ וְאִמְרוּ אֶל-הַשּׁוּעָל הַזֶּה הִנְנִי מְגָרֵשׁ שֵׁדִים וּפֹעֵל רְפוּאוֹת הַיּוֹם וּמָחָר וּבַיּוֹם הַשְּׁלִישִׁי אָבֹא עַד-קִצִּי׃ 33 אֲבָל הָלוֹךְ אֵלֵךְ הַיּוֹם וּמָחָר וּמִמָּחֳרָתוֹ כִּי לֹא-יִתָּכֵן אֲשֶׁר יֹאבַד נָבִיא מִחוּץ לִירוּשָׁלָיִם׃ 34 יְרוּשָׁלַיִם יְרוּשָׁלַיִם הַהֹרֶגֶת אֶת-הַנְּבִיאִים וְהַסֹּקֶלֶת אֶת-הַנִּשְׁלָחִים אֵלֶיהָ (ירמיהו ב' ל') כַּמֶּה פְעָמִים חָפַצְתִּי לְקַבֵּץ אֶת-בָּנַיִךְ כַּאֲשֶׁר תְּקַבֵּץ הַתַּרְנְגֹלֶת אֶת-אֶפְרוֹחֶיהָ תַּחַת כְּנָפֶיהָ וְאַתֶּם לֹא אֲבִיתֶם׃(ישעיהו סה' ב') 35 הִנֵּה בֵיתְכֶם יֵעָזֵב לָכֶם [שָׁמֵם] (ירמיהו יב' ז'; כב' ה') וַאֲנִי אֹמֵר לָכֶם כִּי-רָאֹה לֹא תִרְאוּנִי עַד-בּוֹא הָעֵת אֲשֶׁר תֹּאמְרוּ בָּרוּךְ הַבָּא בְּשֵׁם יְהוָֹה׃(תהילים קיח' כו')
