לפרק הקודם

הבשורה על-פי מתי
פרק כא

ישוע נכנס לירושלים

(מרקוס יא' 11-1;לוקס יט' 38-28;יוחנן יב' 19-12)

1 וַיְהִי כַּאֲשֶׁר קָרְבוּ לִירוּשָׁלַיִם וַיָּבֹאוּ בֵית-פַּגֵי אֶל-הַר הַזֵּיתִים וַיִּשְׁלַח יֵשׁוּעַ שְׁנַיִם מִן-הַתַּלְמִידִים׃ 2 וַיֹּאמֶר אֲלֵיהֶם לְכוּ אֶל-הַכְּפָר אֲשֶׁר מִמוּלְכֶם שָׁם תִּמְצְאוּ אָתוֹן אֲסוּרָה וְעַיִר עִמָּהּ הַתִּירוּ אֹתָם וַהָבִיאוּ אֵלָי׃ 3 וְכִי-יֹאמַר אִישׁ אֲלֵיכֶם דָּבָר וַאֲמַרְתֶּם הָאָדוֹן צָרִיךְ לָהֶם וּמִיָּד יְשַׁלְּחֵם׃ 4 וְכָל-זֹאת הָיָתָה לְמַלּאת מַה-שֶׁנֶּאֱמַר בְּיַד הַנָּבִיא לֵאמֹר׃ 5 אִמְרוּ לְבַת-צִיּוֹן הִנֵּה מַלְכֵּךְ יָבוֹא לָךְ עָנִי וְרֹכֵב עַל-חֲמוֹר וְעַל-עַיִר בֶּן-אֲתֹנוֹת׃(זכריה ט' ט') 6 וַיֵּלְכוּ הַתַּלְמִידִים וַיַּעֲשׂוּ כַּאֲשֶׁר צִוָּה אֹתָם יֵשׁוּעַ׃ 7 וַיָּבִיאוּ אֶת-הָאָתוֹן וְאֶת-הָעַיִר וַיָּשִׂימוּ עֲלֵיהֶם אֶת-בִּגְדֵיהֶם וַיּוֹשִׁיבֻהוּ עֲלֵיהֶם׃ 8 וְרֹב הֶהָמוֹן פָּרְשׂוּ אֶת-בִּגְדֵיהֶם עַל-הַדָּרֶךְ וְאֲחֵרִים כָּרְתוּ סְעִיפֵי עֵצִים וַיִּשְׁטָחוּם עַל-הַדָּרֶךְ׃(מלכים ב', ט' יג') 9 וַהֲמוֹן הָעָם הָהֹלְכִים לְפָנָיו וְאַחֲרָיו קָרְאוּ לֵאמֹר הוֹשַׁע-נָא לְבֶן-דָּוִד בָּרוּךְ הַבָּא בְּשֵׁם יְהוָֹה הוֹשַׁע-נָא בַּמְּרוֹמִים׃(תהילים קיח' כה'-כו') 10 וַיְהִי בְּבֹאוֹ יְרוּשָׁלָיִם וַתֵּהֹם כָּל-הָעִיר וַיֹּאמְרוּ מִי זֶה׃(תהילים כד' ז'-י'; ירמיהו ל' יח'-כב') 11 וַיֹּאמְרוּ הֲמֹן הָעָם זֶה הוּא הַנָּבִיא יֵשׁוּעַ מִנְּצֶרֶת אֲשֶׁר בַּגָּלִיל׃

טיהור בית המקדש

(מרקוס יא' 19-15;לוקס יט' 48-45;יוחנן ב' 22-13)

12 וַיָּבֹא יֵשׁוּעַ אֶל-מִקְדַּשׁ הָאֱלֹהִים וַיְגָרֶשׁ מִשָׁם אֵת כָּל-הַמּוֹכְרִים וְהַקּוֹנִים בַּמִּקְדָּשׁ וַיַּהֲפֹךְ אֶת-שֻׁלְחֲנוֹת הַשֻׁלְחָנִים וְאֶת-מֹשְׁבוֹת מֹכְרֵי הַיּוֹנִים׃(דברים יד' כד'-כו') 13 וַיֹּאמֶר אֲלֵיהֶם הֵן כָּתוּב כִּי בֵיתִי בֵּית תְּפִלָּה יִקָּרֵא (ישעיהו נו' ז') וְאַתֶּם שַׂמְתֶּם אֹתוֹ לִמְעָרַת פָּרִיצִים׃(ירמיהו ז' יא') 14 וַיִּגְּשׁוּ אֵלָיו עִוְרִים וּפִסְחִים בַּמִּקְדָּשׁ וַיִּרְפָּאֵם׃ 15 וַיְהִי כִּרְאוֹת רָאשֵׁי הַכֹּהֲנִים וְהַסּוֹפְרִים אֵת הַנִּפְלָאוֹת אֲשֶׁר עָשָׂה וְאֵת הַיְלָדִים הַצֹּעֲקִים בַּמִּקְדָּשׁ וְאֹמְרִים הוֹשַׁע-נָא לְבֶן-דָּוִד (תהילים קיח' כה') וַיִּחַר לָהֶם׃ 16 וַיֹּאמְרוּ אֵלָיו הֲשֹׁמֵעַ אַתָּה אֶת-אֲשֶׁר אֹמְרִים אֵלֶה וַיֹּאמֶר יֵשׁוּעַ אֲלֵיהֶם שֹׁמֵעַ אָנֹכִי הֲכִי לֹא קְרָאתֶם מִפִּי עוֹלֲלִים וְיוֹנְקִים יִסַּדְתָּ עֹז׃(תהילים ח' ג') 17 וַיַּעַזְבֵם וַיֵּצֵא אֶל-מִחוּץ לָעִיר אֶל-בֵּית-הִינִי וַיָּלֶן שָׁם׃

עץ התאנה שהתייבש

(מרקוס יא' 24-20)

18 וּבַבֹּקֶר שָׁב אֶל-הָעִיר וְהוּא רָעֵב׃ 19 וַיַּרְא תְּאֵנָה אַחַת עַל-הַדָּרֶךְ וַיִּקְרַב אֵלֶיהָ וְלֹא-מָצָא בָהּ מְאוּמָה בִּלְתִּי הֶעָלִים וַיֹּאמֶר אֵלֶיהָ מֵעַתָּה לֹא-יִהְיֶה מִמֵּךְ פְּרִי עַד-עוֹלָם וַתִּיבַשׁ הַתְּאֵנָה פִּתְאֹם׃ 20 וַיִּרְאוּ הַתַּלְמִידִים וַיִּתְמְהוּ וַיֹּאמְרוּ אֵיךְ יָבְשָׁה הַתְּאֵנָה פִּתְאֹם׃ 21 וַיַּעַן יֵשׁוּעַ וַיֹּאמֶר אֲלֵיהֶם אָמֵן אֹמֵר אֲנִי לָכֶם אִם-תִּהְיֶה בָכֶם אֱמוּנָה וְלֹא חָלַק לִבְּכֶם כְּמַעֲשֵׂה הַתְּאֵנָה הַזֹּאת תַּעֲשׂוּ וְאַף תֹּאמְרוּ אֶל-הָהָר הַזֶּה הִנָּשֵׂא וְנֶעְתַּקְתָּ אֶל-תּוֹך הַיָּם וְהָיָה כֵן׃ 22 וְכֹל אֲשֶׁר תִּשְׁאֲלוּ בַּתְּפִלָּה וְאַתֶּם מַאֲמִינִים יָבֹא לָכֶם׃

ישוע משיב לשואלים בעניין סמכותו

(מרקוס יא' 33-27;לוקס כ' 8-1)

23 וַיָּבֹא אֶל-הַמִּקְדָּשׁ וַיְלַמֵּד שָׁם וְרָאשֵׁי הַכֹּהֲנִים וְזִקְנֵי הָעָם נִגְּשׁוּ אֵלָיו וַיֹּאמְרוּ בְּאֵי-זוֹ רְשׁוּת אַתָּה עֹשֶׂה אֵלֶּה וּמִי נָתַן לְךָ הָרְשׁוּת הַזֹּאת׃ 24 וַיַּעַן יֵשׁוּעַ וַיֹּאמֶר אֲלֵיהֶם גַּם-אֲנִי אֶשְׁאֲלָה אֶתְכֶם דָּבָר אֶחָד אֲשֶׁר אִם-תַּגִּידוּ אֹתוֹ לִי גַּם-אֲנִי אַגִּיד לָכֶם בְּאֵי-זוֹ רְשׁוּת אֲנִי עֹשֶׂה אֵלֶּה׃ 25 טְבִילַת יוֹחָנָן מֵאַיִן הָיְתָה הֲמִּשָׁמַיִם אִם-מִבְּנֵי אָדָם וַיַּחְשְׁבוּ בִלְבָבָם לֵאמֹר׃ 26 אִם-נֹאמַר מִשָׁמַיִם וְאָמַר אֵלֵינוּ מַדּוּעַ אֵפוֹא לֹא הֶאֱמַנְתֶּם בּוֹ וְאִם-נֹאמַר מִבְּנֵי אָדָם יְרֵאִים אֲנַחְנוּ אֶת-הֲמוֹן הָעָם כִּי-כֻלָּם חֹשְׁבִים אֶת-יוֹחָנָן לְנָבִיא׃ 27 וַיַּעֲנוּ אֶת-יֵשׁוּעַ וַיֹּאמְרוּ לֹא יָדָעְנוּ וַיֹּאמֶר אֲלֵיהֶם גַּם-אֲנִי לֹא אֹמַר לָכֶם מָה רְשׁוּתִי כִּי אֶעֱשֶׂה אֵלֶּה׃

משל שני הבנים

28 אֲבָל מַה-דַּעְתְּכֶם אִישׁ הָיָה וְלוֹ שְׁנֵי בָנִים וַיִּגַּשׁ אֶל-הָרִאשׁוֹן וַיֹּאמֶר בְּנִי לֵךְ הַיּוֹם וַעֲבֹד בְּכַרְמִי׃ 29 וַיַּעַן וַיֹּאמֶר לֹא חָפַצְתִּי וְאַחֲרֵי-כֵן נִחַם וַיֵּלַךְ׃ 30 וַיִּגַּשׁ אֶל-הַשֵׁנִי וַיְדַבֵּר כָּזֹאת גַּם-אֵלָיו וַיַּעַן וַיֹּאמֶר הִנְנִי אֲדֹנִי וְלֹא הָלָךְ׃ 31 מִי מִשְׁנֵיהֶם עָשָׂה אֶת-רְצוֹן אָבִיו וַיֹּאמְרוּ אֵלָיו הָרִאשׁוֹן וַיֹּאמֶר לָהֶם יֵשׁוּעַ אָמֵן אֹמֵר אֲנִי לָכֶם הַמּוֹכְסִים וְהַזּוֹנוֹת יְקַדְּמוּ אֶתְכֶם לָבוֹא אֶל-מַלְכוּת הָאֱלֹהִים׃ 32 כִּי יוֹחָנָן בָּא אֲלֵיכֶם בְּדֶרֶך צְדָקָה וְלֹא הֶאֱמַנְתֶּם לוֹ וְהַמּוֹכְסִים וְהַזּוֹנוֹת הֵם הֶאֱמִינוּ לוֹ וְאַתֶּם רְאִיתֶם וְלֹא נִחַמְתֶּם אַחֲרֵי-כֵן לְהַאֲמִין לוֹ׃

משל בעל הכרם והחוכרים

(מרקוס יב' 12-1;לוקס כ' 19-9)

33 שִׁמְעוּ מָשָׁל אַחֵר אִישׁ בַּעַל-בַּיִת הָיָה אֲשֶׁר נָטַע כֶּרֶם וַיַּעַשׂ גָּדֵר סָבִיב לוֹ וַיַּחְצֹב יֶקֶב וַיִּבֶן מִגְדָּל בְּתוֹכוֹ וַיִּתְּנֵהוּ אֶל-כֹּרְמִים וַיֵּלֶךְ בְּדֶּרֶךְ מֵרָחוֹק׃ 34 וַיְהִי בְּהַגִּיעַ עֵת הַבָּצִיר וַיִּשְׁלַח עֲבָדָיו אֶל-הַכֹּרְמִים לָקַחַת אֶת-פִּרְיוֹ׃ 35 וַיַּחֲזִיקוּ הַכֹּרְמִים בַּעֲבָדָיו אֶת-זֶה הִכּוּ וְאֶת-זֶה הָרְגוּ וְאֶת-זֶה סָקָלוּ׃(נחמיה ט' כו') 36 וַיּוֹסֶף שְׁלֹחַ עֲבָדִים אֲחֵרִים רַבִּים מִן-הָרִאשׁוֹנִים וְגַם-לָהֶם עֲשׂוּ כֵן׃ 37 וּבָאַחֲרוֹנָה שָׁלַח אֲלֵיהֶם אֶת-בְּנוֹ כִּי אָמַר מִפְּנֵי בְנִי יָגוּרוּ׃ 38 וַיְהִי כִּרְאוֹת הַכֹּרְמִים אֶת-הַבֵּן וַיֹּאמְרוּ אִישׁ אֶל-אָחִיו זֶה הוּא הַיּוֹרֵשׁ לְכוּ וְנַהַרְגֵהוּ וְנֹאחֲזָה בְּנַחֲלָתוֹ׃ 39 וַיַּחֲזִיקוּ בוֹ וַיִּדְחָפוּהוּ אֶל-מִחוּץ לַכָּרֶם וַיַּהַרְגוּ אֹתוֹ׃ 40 וְעַתָּה כִּי-יָבוֹא בַּעַל הַכָּרֶם מַה-יַּעֲשֶׂה לַכֹּרְמִים הָהֵם׃ 41 וַיֹּאמְרוּ אֵלָיו יָרַע לָרָעִים וִיאַבְּדֵם וְאֶת-הַכֶּרֶם יִתֵּן לְכֹרְמִים אֲחֵרִים אֲשֶׁר יָשִׁיבוּ לוֹ אֶת-פִּרְיוֹ בְּעִתּוֹ׃ 42 וַיֹּאמֶר אֲלֵיהֶם יֵשׁוּעַ הֲכִי לֹא-קְרָאתֶם בַּכְּתוּבִים אֶבֶן מָאֲסוּ הַבּוֹנִים הָיְתָה לְרֹאשׁ פִּנָּה מֵאֵת יְהוָֹה הָיְתָה זֹּאת הִיא נִפְלָאת בְּעֵינֵינוּ׃(תהילים קיח' כב'-כג') 43 עַל-כֵּן אֲנִי אֹמֵר לָכֶם כִּי-תֻקַּח מִכֶּם מַלְכוּת הָאֱלֹהִים וְתִסֹּב לְגוֹי אֲשֶׁר יַעֲשֶׂה אֶת-פִּרְיָהּ: [ 44 וְהַנֹּפֵל אַל-הָאֶבֶן הַהִיא יִשָׁבֵר (ישעיהו ח' יד'-טו') וַאֲשֶׁר תִּפֹּל עָלָיו תִּשְׁחָקֵהוּ:] (דניאל ב' מב') 45 וַיְהִי כִּשְׁמֹעַ רָאשֵׁי הַכֹּהֲנִים וְהַפְּרוּשִׁים אֶת-מְשָׁלָיו וַיָּבִינוּ כִּי עֲלֵיהֶם דִּבֵּר: 46 וַיְבַקְשׁוּ לְתָפְשׂוֹ וַיִּירְאוּ מִפְּנֵי רָב-הָעָם כִּי לְנָבִיא חֲשָׁבֻהוּ:
