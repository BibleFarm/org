אגרת שאול אל האפסים

מן המוות אל החיים

1 גַּם-אֶתְכֶם אֲשֶׁר מֵתִים הֱיִיתֶם בְּפִשְׁעֵיכֶם וְחַטֹּאתֵיכֶם׃ 2 אֲשֶׁר הִתְהַלַּכְתֶּם בָּהֶם כְּדוֹר הָעוֹלָם הַזֶּה כִּרְצוֹן שַׂר מֶמְשֶׁלֶת הָאֲוִיר וְהוּא הָרוּחַ הַפֹּעֵל כָּעֵת בִּבְנֵי הַמֶּרִי׃ 3 וְגַם-אֲנַחְנוּ כֻלָּנוּ בְּתוֹכָם הָלַכְנוּ לְפָנִים בְּתַאֲוֹת הַבָּשָׂר לַעֲשׂוֹת חֶפְצֵי בְשָׂרֵנוּ וּמַחְשְׁבוֹתֵינוּ וַנְּהִי בְנֵי-רֹגֶז בְּטִבְעֵנוּ כִּשְׁאָר בְּנֵי אָדָם׃ 4 אֲבָל-הָאֱלֹהִים הַמָּלֵא רַחֲמִים (תהילים קג' ח') בְּרֹב אַהֲבָתוֹ אֲשֶׁר אָהַב אֹתָנוּ׃ 5 אַחֲרֵי הֱיוֹתֵנוּ מֵתִים בַּפְּשָׁעִים הֶחֱיָנוּ עִם-הַמָּשִׁיחַ בַּחֶסֶד נוֹשַׁעְתֶּם׃ 6 וַיְעִירֵנוּ אִתּוֹ אַף-הוֹשִׁיבָנוּ בַּמְּרוֹמִים בַּמָּשִׁיחַ יֵשׁוּעַ׃ 7 לְהַרְאוֹת בַּדֹּרוֹת הַבָּאִים אֶת-גְּדֻלַּת עשֶׁר חַסְדּוֹ בְּטוֹבָתוֹ עָלֵינוּ בַּמָּשִׁיחַ יֵשׁוּעַ׃ 8 כִּי-בַחֶסֶד נוֹשַׁעְתֶּם עַל-יְדֵי הָאֱמוּנָה וְלֹא מִיֶּדְכֶם הָיְתָה זֹּאת כִּי-מַתַּת אֱלֹהִים הִיא׃ 9 לֹא מִתּוֹךְ הַמַּעֲשִׂים שֶׁלּא יִתְהַלֵּל אִישׁ׃ 10 כִּי-פֹעַל אֱלֹהִים אֲנַחְנוּ נִבְרָאִים (תהילים ק' ג') בַּמָּשִׁיחַ יֵשׁוּעַ לְמַעֲשִׂים טוֹבִים אֲשֶׁר הֵכִין הָאֱלֹהִים מִקֶּדֶם לְמַעַן נִתְהַלֵּךְ בָּהֶם׃

אחדות במשיח

11 עַל-כֵּן זִכְרוּ כִּי-אַתֶּם מִתְּחִלָּה הַגּוֹיִם בַּבָּשָׂר הַנִּקְרָאִים עֲרֵלִים בְּפִי הַנִּקְרָאִים בְּנֵי הַמִּילָה שֶׁהִיא מַעֲשֵׂה יָדַיִם בַּבָּשָׂר׃ 12 כִּי-אַתֶּם בָּעֵת הַהִיא הֱיִיתֶם בְּלִי מָשִׁיחַ זָרִים מֵעֲדַת יִשְׂרָאֵל וְנָכְרִים לִבְרִיתוֹת הַהַבְטָחָה (שמות יט' ה'-ו'; ירמיהו לא' לא'-לב') בְּאֵין תִּקְוָה וּבְאֵין לָכֶם אֱלֹהִים בָּעוֹלָם׃ 13 וְעַתָּה בְּיֵשׁוּעַ הַמָּשִׁיחַ אַתֶּם הָרְחוֹקִים מֵאָז הֱיִיתֶם קְרוֹבִים בְּדַם הַמָּשִׁיחַ׃(תהילים קמח' יד') 14 כִּי הוּא שְׁלוֹמֵנוּ אֲשֶׁר עָשָׂה הַשְּׁנַיִם אֶחָד וְהָרַס מְחִיצַת הַגָּדֵר׃ 15 בְּבַטְּלוֹ הָאֵיבָה בִּבְשָׂרוֹ אֶת-תּוֹרַת הַמִּצְוֹת וְהַחֻקּוֹת לִבְרֹא בְנַפְשׁוֹ אֶת-הַשְּׁנַיִם לְאָדָם אֶחָד חָדָשׁ וַיַּעַשׂ שָׁלוֹם׃ 16 וַיְרַצֶּה אֶת-שְׁנֵיהֶם בְּגוּף אֶחָד לֵאלֹהִים עַל-יְדֵי צְלִיבָתוֹ בַּהֲמִיתוֹ בְנַפְשׁוֹ אֶת-הָאֵיבָה׃ 17 וַיָּבֹא וַיְבַשֵׂר לָכֶם שָׁלוֹם שָׁלוֹם לָרָחוֹק וְלַקָּרוֹב׃(ישעיהו נז' יט') 18 כִּי עַל-יָדוֹ יֵשׁ לִשְׁנֵינוּ מָבוֹא בְּרוּחַ אֶחָד אֶל-אָבִינוּ׃ 19 לָכֵן אֵינְכֶם עוֹד גֵּרִים וְתוֹשָׁבִים כִּי-בְנֵי-עִירָם שֶׁל-הַקְּדשִׁים וּבְנֵי בֵּית אֱלֹהִים אַתֶּם׃ 20 בְּנוּיִם עַל-יְסוֹד הַשְּׁלִיחִים וְהַנְּבִיאִים וְיֵשׁוּעַ הַמָּשִׁיחַ הוּא אֶבֶן הַפִּנָּה׃(תהילים קיח' כב'; ישעיהו כח' טז') 21 אֲשֶׁר חֻבַּר-בּוֹ יַחַד הַבִּנְיָן כֻּלּוֹ עַד אֲשֶׁר-יִגְבַּהּ לְהֵיכַל קֹדֶשׁ לַיהוָֹה׃ 22 וּבוֹ נִבְנִים גַּם-אַתֶּם לִהְיוֹת מִשְׁכַּן אֱלֹהִים בָּרוּחַ׃
